12  package com.dqgb.sshframe.common.util;
13  
14  import com.dqgb.cordys.sso.CordysSSO;
15  
16  public class CordysUtil {
17      public static String[] getCordysParam()
18          throws Exception {
19          String cordysUrl = PropertyUtil.get("cordys.url");
20          String cordysOrganization = PropertyUtil.get("cordys.organization.dn");
21          String username = PropertyUtil.get("cordys.username");
22          String password = PropertyUtil.get("cordys.password");
23          String[] params = {cordysUrl, cordysOrganization, username, password};
24          
25          CordysSSO sso = new CordysSSO(params);
26          params[2] = sso.getTokenByUsername(params[2], params[3]);
27          return params;
28      }
29  }
