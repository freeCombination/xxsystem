124      * <p>
125      * Description: LDAPFastBind
126      * </p>
127      * 
128      * @param ldapurl
129      */
130     public LDAPFastBind(String ldapurl) {
131         env = new Hashtable();
132         env.put(Context.INITIAL_CONTEXT_FACTORY,
133             "com.sun.jndi.ldap.LdapCtxFactory");
134         env.put(Context.SECURITY_AUTHENTICATION, "simple");
135         env.put(Context.PROVIDER_URL, ldapurl);
136         
137         env.put(Context.SECURITY_PROTOCOL, "ssl");
138         
139         String keystore =
140             "J:/Program Files/Java/jdk1.6.0_17/jre/lib/security/cacerts";
141         System.setProperty("javax.net.ssl.trustStore", keystore);
142         
143         connCtls = new Control[] {new FastBindConnectionControl()};
144         
145         try {
146             ctx = new InitialLdapContext(env, connCtls);
147             
148         }
149         catch (NamingException e) {
150         }
151     }

153     /**
154      * LDAPFastBind
155      * 
156      * @Title Authenticate
157      * @author wanglc
158      * @date 2013-12-6
159      * @param username
160      * @param password
161      * @return
162      */
163     public boolean Authenticate(String username, String password) {
164         try {
165             ctx.addToEnvironment(Context.SECURITY_PRINCIPAL, username);
166             ctx.addToEnvironment(Context.SECURITY_CREDENTIALS, password);
167             ctx.reconnect(connCtls);
168             return true;
169         }
170         catch (AuthenticationException e) {
171             return false;
172         }
173         catch (NamingException e) {
174             return false;
175         }
176     }

195      * printUserAccountControl
196      * 
197      * @Title printUserAccountControl
198      * @author wanglc
199      * @date 2013-12-6
200      */
201     public void printUserAccountControl() {
202         try {
203             
204             // Create the search controls
205             SearchControls searchCtls = new SearchControls();
206             
207             // Specify the search scope
208             searchCtls.setSearchScope(SearchControls.SUBTREE_SCOPE);
209             
210             // specify the LDAP search filter
211             // String searchFilter = "(&(objectClass=user)(CN=test))";
212             // String searchFilter = "(&(objectClass=group))";
213             String searchFilter = "(&(objectClass=user)(CN=peter lee))";
214             
215             // Specify the Base for the search
216             String searchBase = "DC=joeyta,DC=local";
217             
218             // initialize counter to total the group members
219             int totalResults = 0;
220             
221             // Specify the attributes to return
222             String returnedAtts[] = {"givenName", "mail"};
223             searchCtls.setReturningAttributes(returnedAtts);
224             
225             // Search for objects using the filter
226             NamingEnumeration answer =
227                 ctx.search(searchBase, searchFilter, searchCtls);
228             
229             // Loop through the search results
230             while (answer.hasMoreElements()) {
231                 SearchResult sr = (SearchResult)answer.next();
232                 
233                 // Print out the groups
234                 
235                 Attributes attrs = sr.getAttributes();
236                 if (attrs != null) {
237                     
238                     try {
239                         for (NamingEnumeration ae = attrs.getAll(); ae.hasMore();) {
240                             Attribute attr = (Attribute)ae.next();
241                             for (NamingEnumeration e = attr.getAll(); e.hasMore(); totalResults++) {
242                             }
243                             
244                         }
245                         
246                     }
247                     catch (NamingException e) {
248                         System.err.println("Problem listing membership: " + e);
249                     }
250                     
251                 }
252             }
253             
254         }
255         
256         catch (NamingException e) {
257             System.err.println("Problem searching directory: " + e);
258         }
259         
260     }

262     /**
263      * adminChangePassword
264      * 
265      * @Title adminChangePassword
266      * @author wanglc
267      * @date 2013-12-6
268      * @param sUserName
269      * @param sNewPassword
270      * @return
271      */
272     public boolean adminChangePassword(String sUserName, String sNewPassword) {
273         try {
274             // set password is a ldap modfy operation
275             ModificationItem[] mods = new ModificationItem[1];
276             
277             // Replace the "unicdodePwd" attribute with a new value
278             // Password must be both Unicode and a quoted string
279             String newQuotedPassword = "\"" + sNewPassword + "\"";
280             byte[] newUnicodePassword = newQuotedPassword.getBytes("UTF-16LE");
281             
282             mods[0] =
283                 new ModificationItem(DirContext.REPLACE_ATTRIBUTE,
284                     new BasicAttribute("unicodePwd", newUnicodePassword));
285             
286             // Perform the update
287             ctx.modifyAttributes(sUserName, mods);
288             
289             return true;
290         }
291         catch (NamingException e) {
292         }
293         catch (UnsupportedEncodingException e) {
294         }
295         return false;
296     }

298     /**
299      * userChangePassword
300      * 
301      * @Title userChangePassword
302      * @author wanglc
303      * @date 2013-12-6
304      * @param sUserName
305      * @param sOldPassword
306      * @param sNewPassword
307      * @return
308      */
309     public boolean userChangePassword(String sUserName, String sOldPassword,
310         String sNewPassword) {
311         try {
312             // StartTlsResponse tls =
313             // (StartTlsResponse)ctx.extendedOperation(new StartTlsRequest());
314             // tls.negotiate();
315             
316             // change password is a single ldap modify operation
317             // that deletes the old password and adds the new password
318             ModificationItem[] mods = new ModificationItem[2];
319             
320             // Firstly delete the "unicdodePwd" attribute, using the old
321             // password
322             // Then add the new password,Passwords must be both Unicode and a
323             // quoted string
324             String oldQuotedPassword = "\"" + sOldPassword + "\"";
325             byte[] oldUnicodePassword = oldQuotedPassword.getBytes("UTF-16LE");
326             String newQuotedPassword = "\"" + sNewPassword + "\"";
327             byte[] newUnicodePassword = newQuotedPassword.getBytes("UTF-16LE");
328             
329             mods[0] =
330                 new ModificationItem(DirContext.REMOVE_ATTRIBUTE,
331                     new BasicAttribute("unicodePwd", oldUnicodePassword));
332             mods[1] =
333                 new ModificationItem(DirContext.ADD_ATTRIBUTE,
334                     new BasicAttribute("unicodePwd", newUnicodePassword));
335             
336             // Perform the update
337             ctx.modifyAttributes(sUserName, mods);
338             
339             // tls.close();
340             return true;
341             
342         }
343         catch (NamingException e) {
344             System.err.println("Problem changing password: " + e);
345         }
346         catch (UnsupportedEncodingException e) {
347             System.err.println("Problem encoding password: " + e);
348         }
349         catch (Exception e) {
350             System.err.println("Problem: " + e);
351         }
352         return false;
353     }

355     /**
356      * createNewUser
357      * 
358      * @Title createNewUser
359      * @author wanglc
360      * @date 2013-12-6
361      * @param sGroupName
362      * @param sUserName
363      * @return
364      */
365     public boolean createNewUser(String sGroupName, String sUserName) {
366         try {
367             // Create attributes to be associated with the new user
368             Attributes attrs = new BasicAttributes(true);
369             
370             // These are the mandatory attributes for a user object
371             // Note that Win2K3 will automagically create a random
372             // samAccountName if it is not present. (Win2K does not)
373             attrs.put("objectClass", "user");
374             attrs.put("sAMAccountName", "AlanT");
375             attrs.put("cn", "Alan Tang");
376             
377             // These are some optional (but useful) attributes
378             attrs.put("givenName", "Alan");
379             attrs.put("sn", "Tang");
380             attrs.put("displayName", "Alan Tang");
381             attrs.put("description", "Engineer");
382             attrs.put("userPrincipalName", "alan-AT-joeyta.local");
383             attrs.put("mail", "alang-AT-mail.joeyta-DOT-local");
384             attrs.put("telephoneNumber", "123 456 789");
385             
386             // some useful constants from lmaccess.h
387             int UF_ACCOUNTDISABLE = 0x0002;
388             int UF_PASSWD_NOTREQD = 0x0020;
389             int UF_NORMAL_ACCOUNT = 0x0200;
390             int UF_PASSWORD_EXPIRED = 0x800000;
391             
392             attrs.put("userAccountControl",
393                 Integer.toString(UF_NORMAL_ACCOUNT + UF_PASSWD_NOTREQD
394                     + UF_PASSWORD_EXPIRED + UF_ACCOUNTDISABLE));
395             
396             ModificationItem[] mods = new ModificationItem[2];
397             String newQuotedPassword = "\"P-AT-ssw0rd\"";
398             byte[] newUnicodePassword = newQuotedPassword.getBytes("UTF-16LE");
399             
400             mods[0] =
401                 new ModificationItem(DirContext.REPLACE_ATTRIBUTE,
402                     new BasicAttribute("unicodePwd", newUnicodePassword));
403             mods[1] =
404                 new ModificationItem(DirContext.REPLACE_ATTRIBUTE,
405                     new BasicAttribute("userAccountControl",
406                         Integer.toString(UF_NORMAL_ACCOUNT
407                             + UF_PASSWORD_EXPIRED)));
408             
409             ctx.modifyAttributes(sUserName, mods);
410             
411             try {
412                 ModificationItem member[] = new ModificationItem[1];
413                 member[0] =
414                     new ModificationItem(DirContext.ADD_ATTRIBUTE,
415                         new BasicAttribute("member", sUserName));
416                 
417                 ctx.modifyAttributes(sGroupName, member);
418                 
419             }
420             catch (NamingException e) {
421                 System.err.println("Problem adding user to group: " + e);
422             }
423             // Could have put tls.close() prior to the group modification
424             // but it seems to screw up the connection or context
425             
426             return true;
427             
428         }
429         catch (NamingException e) {
430             System.err.println("Problem creating object: " + e);
431         }
432         
433         catch (IOException e) {
434             System.err.println("Problem creating object: " + e);
435         }
436         return false;
437     }
