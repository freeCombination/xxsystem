18  import org.springframework.beans.factory.annotation.Qualifier;
19  
20  import com.dqgb.sshframe.common.util.JsonUtil;
21  import com.dqgb.sshframe.common.util.RequestUtil;

26  /**
27   * 系统拦截器，拦截所有的方法
28   * 
29   * @author ndy
30   * @version V1.20,2014年2月13日 上午10:38:57
31   * @see [相关类/方法]
32   * @since V1.20
33   * @depricated
34   */
35  public class MethodInterceptor extends AbstractInterceptor {
36      
37      /**
38       * @Fields serialVersionUID :
39       */
40      private static final long serialVersionUID = -2163314983372693453L;
41      
42      @Autowired(required = true)
43      @Qualifier("loginAction")
44      LoginAction loginAction;
45      
46      /**
47       * <p>
48       * Title intercept
49       * </p>
50       * <p>
51       * Author ndy
52       * </p>
53       * <p>
54       * Description
55       * </p>
56       * 
57       * @param arg0
58       * @throws Exception
59       * @see com.opensymphony.xwork2.interceptor.AbstractInterceptor#intercept(com.opensymphony.xwork2.ActionInvocation)
60       */
61      @Override
62      public String intercept(ActionInvocation arg0)
63          throws Exception {
64          HttpServletRequest request = ServletActionContext.getRequest();
65          String ssotoken = request.getParameter("SSOTOKEN");
66          HttpSession session = request.getSession();
67  		if (session.getAttribute("currentUserName") == null && StringUtils.isNotBlank(ssotoken) ) {
68  			loginAction.setSSOTOKEN(ssotoken);
69  			loginAction.login();
70  		}
71          if (session.getAttribute("currentUserName") == null) {
72              String requestType = request.getHeader("X-Requested-With");
73              if(StringUtils.isNotBlank(requestType) && requestType.equalsIgnoreCase("XMLHttpRequest")){
74                  HttpServletResponse response = ServletActionContext.getResponse();
75                  JsonUtil.outJson("{'sessionstatus':'true'}");
76                  response.setHeader("sessionstatus", "timeout");
77                  return null;
78              }
79              else {
80                  String url = RequestUtil.getProjectURL(request);
81                  String str =
82                      "<script type=\"text/javascript\">top.window.location.href='"
83                          + url + "';</script>";// 页面接收
84                  HttpServletResponse response = ServletActionContext.getResponse();
85                  response.setContentType("text/html; charset=utf-8");
86                  response.setHeader("Pragma", "No-cache");
87                  response.setHeader("Cache-Control", "no-cache");
88                  response.setDateHeader("Expires", 0);
89                  response.getWriter().write(str);
90                  response.getWriter().flush();
91                  response.getWriter().close();
92                  return null;
93              }
94          }
95          else {
96              return arg0.invoke();
97          }
98      }
