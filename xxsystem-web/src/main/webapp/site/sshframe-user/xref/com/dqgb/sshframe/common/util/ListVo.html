12  package com.dqgb.sshframe.common.util;
13  
14  import java.util.ArrayList;
15  import java.util.List;
16  
17  /**
18   * 封装一个List及其总条目
19   * 
20   * @author wanglc
21   * @version V1.20,2013-12-6 下午3:06:23
22   * @see [相关类/方法]
23   * @since V1.20
24   * @depricated
25   */
26  public class ListVo<T> {
27      
28      /** @Fields totalSize : 总条数 */
29      private int totalSize;
30      
31      /** @Fields data : 数据集合 */
32      private List<T> data;
33      
34      /**
35       * <p>
36       * Title: ListVo
37       * </p>
38       * <p>
39       * Description: 默认构造方法
40       * </p>
41       */
42      public ListVo() {
43          this.totalSize = 0;
44          data = new ArrayList<T>();
45      }
46      
47      /**
48       * <p>
49       * Title: ListVo
50       * </p>
51       * <p>
52       * Description: full构造方法
53       * </p>
54       * 
55       * @param totalSize 总条数
56       * @param data 数据集合
57       */
58      public ListVo(int totalSize, List<T> data) {
59          this.totalSize = totalSize;
60          this.data = data;
61      }
62      
63      public List<T> getData() {
64          return data;
65      }
66      
67      public void setData(List<T> data) {
68          this.data = data;
69      }
70      
71      public int getTotalSize() {
72          return totalSize;
73      }
74      
75      public void setTotalSize(int totalSize) {
76          this.totalSize = totalSize;
77      }
78  }
