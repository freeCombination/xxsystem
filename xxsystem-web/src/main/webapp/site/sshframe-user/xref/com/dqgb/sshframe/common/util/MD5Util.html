27  @SuppressWarnings("unused")
28  public abstract class MD5Util {
29      private static final Logger log = Logger.getLogger(MD5Util.class);
30      
31      /** @Fields hexDigits : 16进制数 */
32      private final static String[] hexDigits = {"0", "1", "2", "3", "4", "5",
33          "6", "7", "a", "b", "c", "d", "e", "f", "8", "9"};
34      
35      /**
36       * 转换字节数组为16进制字串
37       * 
38       * @param b 字节数组
39       * @return 16进制字串
40       */
41      private static String byteArrayToHexString(byte[] b) {
42          StringBuffer resultSb = new StringBuffer();
43          for (int i = 0; i < b.length; i++) {
44              resultSb.append(byteToHexString(b[i]));
45          }
46          return resultSb.toString();
47      }
48      
49      /**
50       * byteToHexString
51       * 
52       * @Title byteToHexString
53       * @author wanglc
54       * @Description:
55       * @date 2013-12-6
56       * @param b
57       * @return
58       */
59      private static String byteToHexString(byte b) {
60          int n = b;
61          if (n < 0)
62              n = 256 + n;
63          int d1 = n / 16;
64          int d2 = n % 16;
65          return hexDigits[d1] + hexDigits[d2];
66      }

85      /**
86       * MD5加密
87       * 
88       * @Title encode
89       * @author wanglc
90       * @Description:
91       * @date 2014-2-13
92       * @param plan
93       * @return
94       */
95      public static String encode(String plan) {
96          String ret;
97          try {
98              MessageDigest md = MessageDigest.getInstance("MD5");
99              md.update(plan.getBytes());
100             byte b[] = md.digest();
101             int i;
102             StringBuffer buf = new StringBuffer("");
103             for (int offset = 0; offset < b.length; offset++) {
104                 i = b[offset];
105                 if (i < 0)
106                     i += 256;
107                 if (i < 16)
108                     buf.append("0");
109                 buf.append(Integer.toHexString(i));
110             }
111             ret = buf.toString();
112         }
113         catch (Exception e) {
114             log.error(" ExcelUtil文件中【导出excel】方法发生异常：" + e);
115             ret = "错误" + e.getMessage();
116         }
117         return ret;
118     }
