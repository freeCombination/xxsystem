11  import java.io.IOException;
12  import java.io.InputStream;

14  import java.security.KeyFactory;
15  import java.security.NoSuchAlgorithmException;
16  import java.security.PrivateKey;
17  import java.security.PublicKey;
18  import java.security.interfaces.RSAPrivateKey;
19  import java.security.spec.InvalidKeySpecException;
20  import java.security.spec.PKCS8EncodedKeySpec;
21  import java.security.spec.X509EncodedKeySpec;
22  import java.util.Properties;
23  
24  import javax.crypto.Cipher;
25  
26  import org.bouncycastle.jce.provider.BouncyCastleProvider;
27  
28  import com.sun.org.apache.xerces.internal.impl.dv.util.Base64;

30  /**
31   * 加/解密工具类
32   * 
33   * @author wanglc
34   * @version V1.20,2013-12-6 下午3:25:28
35   * @see [相关类/方法]
36   * @since V1.20
37   * @depricated
38   */
39  @SuppressWarnings("restriction")
40  public class RsaCryptoUtil {
41      /** @Fields PUB_KEY : PUB_KEY 公钥 */
42      public static final String PUB_KEY = "pub_key";
43      
44      /** @Fields PRIV_KEY : PRIV_KEY 私钥 */
45      public static final String PRIV_KEY = "priv_key";
46      
47      /**
48       * @Title loadKey
49       * @author wanglc
50       * @Description: 加载密钥
51       * @date 2013-12-6
52       * @param type
53       * @return
54       * @throws NoSuchAlgorithmException
55       * @throws InvalidKeySpecException
56       * @throws IOException
57       */
58      public static Key loadKey(String type)
59          throws NoSuchAlgorithmException, InvalidKeySpecException, IOException {
60          InputStream is =
61              RsaCryptoUtil.class.getResourceAsStream("/application.properties");
62          Properties config = new Properties();
63          config.load(is);
64          KeyFactory keyFactory =
65              KeyFactory.getInstance("RSA", new BouncyCastleProvider());
66          
67          if (PRIV_KEY.equals(type)) {
68              // privateKey
69              String privateKeyValue = config.getProperty(PRIV_KEY);
70              PKCS8EncodedKeySpec priPKCS8 =
71                  new PKCS8EncodedKeySpec(Base64.decode(privateKeyValue));
72              PrivateKey privateKey = keyFactory.generatePrivate(priPKCS8);
73              return privateKey;
74              
75          } else {
76              // publicKey
77              String privateKeyValue = config.getProperty(PUB_KEY);
78              X509EncodedKeySpec bobPubKeySpec =
79                  new X509EncodedKeySpec(Base64.decode(privateKeyValue));
80              PublicKey publicKey = keyFactory.generatePublic(bobPubKeySpec);
81              return publicKey;
82          }
83      }
84      
85      /**
86       * @Title decrypt
87       * @author wanglc
88       * @Description: decrypt
89       * @date 2013-12-6
90       * @param privateKey
91       * @param cipherText
92       * @return
93       */
94      public static String decrypt(RSAPrivateKey privateKey, String cipherText) {
95          byte[] raw = Base64.decode(cipherText);
96          if (privateKey != null) {
97              try {
98                  Cipher cipher =
99                      Cipher.getInstance("RSA", new BouncyCastleProvider());
100                 cipher.init(Cipher.DECRYPT_MODE, privateKey);
101                 return new String(cipher.doFinal(raw));
102             } catch (Exception e) {
103                 throw new RuntimeException("解密失败！");
104             }
105         }
106         return null;
107     }
