68      @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
69      @Override

71          throws BusinessException {
72          // 取得字典类型的uuid
73          log.setLogTypeUUID(log.getType().getFkDictTypeUUID());

75          this.baseDao.save(log);

78      /**
79       * 查询日志
80       * 
81       * @Title getLogList
82       * @author wanglc
83       * @Description:
84       * @date 2013-11-25
85       * @param params 参数Map
86       * @return ListVo<Log> 日志列表集合
87       */
88      @Override
89      public ListVo<LogVo> getLogList(Map<String, String> params)
90          throws BusinessException {
91          // 获取页面参数
92          String startTime = params.get("startDate");
93          String endTime = params.get("endDate");
94          String type = params.get("type");
95          String logContent = params.get("logContent");
96          int start = NumberUtils.toInt(params.get("start"));
97          int limit = NumberUtils.toInt(params.get("limit"));
98          ListVo<LogVo> vo = new ListVo<LogVo>();
99          List<LogVo> voList = new ArrayList<LogVo>();
100         Map pram = new HashMap();
101         // 新建查询HQL语句
102         StringBuffer logHql = new StringBuffer("from Log l ");
103         logHql.append(" where l.type.pkDictionaryId = ");
104         logHql.append(type);
105         logHql.append(StringUtils.isNotBlank(logContent) ? 
106             " and l. opContent like '%" + logContent + "%'" : "");
107         if(StringUtils.isNotBlank(startTime)){
108         	pram.put("startTime", DateUtil.string2date(startTime));
109         	logHql.append(StringUtils.isNotBlank(startTime) ? 
110         	            " and l.opDate >= :startTime" : "");
111         }
112         if(StringUtils.isNotBlank(endTime)){
113         	Date d = new Date(DateUtil.string2date(endTime).getTime()+24*60*60*1000);
114         	pram.put("endTime", d);
115         	logHql.append(StringUtils.isNotBlank(endTime) ? 
116                     " and l.opDate<:endTime" : "");
117         }
118         
119         // 查询总条数
120         int totalCount = baseDao.queryTotalCount(logHql.toString(), pram);
121         
122         logHql.append(" order by l.opDate desc ");
123         // queryEntitysByPage(start, limit, hql, new HashMap());
124         // 分页查询
125         List<Log> logList =
126             (List<Log>)baseDao.findPageByQuery(start,
127                 limit,
128                 logHql.toString(),
129                 pram);
130         
131         // entity转换为vo
132         if (logList != null && logList.size() > 0) {
133             for (Log l : logList) {
134                 LogVo lv = new LogVo(l);
135                 voList.add(lv);
136             }
137         }
138         // 设置返回结果
139         vo.setList(voList);
140         vo.setTotalSize(totalCount);
141         return vo;
142     }

144     /**
145      * 删除日志
146      * 
147      * @Title delLog
148      * @author wanglc
149      * @Description:
150      * @date 2013-11-25
151      * @param params 参数Map
152      * @return String 删除结果
153      */
154     @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
155     @Override
156     public String delLog(Map<String, String> params)
157         throws BusinessException {
158         String ids = params.get("ids");
159         if (ids == null || "".equals(ids)) {
160             return "{success:'false',msg:'删除日志失败'}";
161         }
162         else {
163             for (String logId : ids.split(",")) {
164                 baseDao.deleteEntityById(Log.class,
165                     NumberUtils.toLong(logId));
166             }
167         }
168         return "{success:'true',msg:'删除日志成功'}";
169     }

183         throws BusinessException {

185         return baseDao.queryEntitys(" from  Dictionary d  where d.dictionaryType.typeCode='"
