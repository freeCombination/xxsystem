3   import java.lang.annotation.Documented;

5   import java.lang.annotation.Inherited;
6   import java.lang.annotation.Retention;

10  /**
11   * 类的方法描述注解
12   * 
13   * @author LuoYu
14   */
15  @Target(ElementType.METHOD)
16  @Retention(RetentionPolicy.RUNTIME)
17  @Documented
18  @Inherited
19  public @interface Log {
20      
21      /** 要执行的操作类型比如：add操作 **/
22      public String operationType() default "";
23      
24      /** 要执行的具体操作比如：【添加仓库】 **/
25      public String operationName() default "";
26      
27      /** 实体位置 **/
28      public int position() default 1;
29  }
