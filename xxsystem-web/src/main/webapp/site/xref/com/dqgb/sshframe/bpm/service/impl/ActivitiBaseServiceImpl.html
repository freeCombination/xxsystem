206     /**
207      * 发布流程定义xml文件
208      * 
209      * @author 张小虎
210      * @param absolutePath文件绝对路径
211      * @throw ActivitiException
212      * @return String
213      */
214     @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
215     public String deployProcessDefinitionByXml(String absolutePath)
216         throws ActivitiException
217     {
218         
219         InputStream inputStream = null;
220         File file = new File(absolutePath);
221         try
222         {
223             synchronized (absolutePath)
224             {
225                 inputStream = new FileInputStream(file);
226                 String deployMentFileName = file.getPath();
227                 if (file.getPath().lastIndexOf("\\") != -1)
228                 {
229                     deployMentFileName =
230                         file.getPath().substring(file.getPath()
231                             .lastIndexOf("\\") + 1);
232                 }
233                 else if (file.getPath().lastIndexOf("/") != -1)
234                 {
235                     deployMentFileName =
236                         file.getPath().substring(file.getPath()
237                             .lastIndexOf("/") + 1);
238                 }
239                 DeploymentBuilder d =
240                     repositoryService.createDeployment()
241                         .addInputStream(deployMentFileName, inputStream);
242                 inputStream.close();
243                 
244                 return d.deploy().getId();
245             }
246             
247         }
248         catch (FileNotFoundException e)
249         {
250             throw new ActivitiException(file.getAbsolutePath() + ",文件不存在",e);
251         }
252         catch (Exception e)
253         {
254             throw new ActivitiException("发布失败！" + e.getMessage(),e);
255         }
256         
257     }

461     /**
462      * 回退任务
463      * 
464      * @author 张小虎
465      * @param taskId 任务id
466      * @param activityId 需要跳转活动节点Id
467      * @throw ActivitiException
468      * @return void
469      */
470     @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
471     public void rollbackTask(String taskId, String activityId, Map commitMap)
472         throws ActivitiException
473     {
474         Task task = getTask(taskId);
475         if (task == null)
476         {
477             throw new ActivitiException("任务不存在");
478         }
479         
480         ProcessDefinitionEntity processDefinition =
481             (ProcessDefinitionEntity)((RepositoryServiceImpl)repositoryService).
482             getDeployedProcessDefinition(task.getProcessDefinitionId());
483         ActivityImpl activityImpl =
484             processDefinition.findActivity(task.getTaskDefinitionKey());
485         List<PvmTransition> pvmOrgTransitions =
486             activityImpl.getOutgoingTransitions();
487         List<PvmTransition> pvmTransitions = new ArrayList<PvmTransition>();
488         for (PvmTransition pv : pvmOrgTransitions)
489         {
490             pvmTransitions.add(pv);
491         }
492         // 清除当前节点的流出
493         pvmOrgTransitions.removeAll(pvmOrgTransitions);
494         TransitionImpl newTransition = activityImpl.createOutgoingTransition();
495         // 设置到需要跳转的节点
496         ActivityImpl newActivityImpl =
497             processDefinition.findActivity(activityId);
498         newTransition.setDestination(newActivityImpl);
499         
500         // 提交当前任务，任务跳转到rollTaskId的节点上去
501         if (commitMap != null && commitMap.size() > 0)
502         {
503             completeTask(taskId, commitMap);
504         }
505         else
506         {
507             completeTask(taskId, null);
508         }
509         
510         // 获取更改后的当前节点流出
511         pvmOrgTransitions = activityImpl.getOutgoingTransitions();
512         // 清除掉更改的当前节点的流出
513         pvmOrgTransitions.remove(newTransition);
514         // 恢复原来的流出
515         pvmOrgTransitions.addAll(pvmTransitions);
516         // 去掉转向节点的输入
517         List<PvmTransition> desTransitions =
518             newActivityImpl.getIncomingTransitions();
519         
520         desTransitions.remove(newTransition);
521     }

563     /**
564      * 返回当前正在处理的活动节点
565      * 
566      * @author 张小虎
567      * @param taskId 任务Id
568      * @throw ActivitiException
569      * @return List<ActivityImpl>
570      */
571     
572     public List<ActivityImpl> getActivityImpl(String taskId)
573         throws ActivitiException
574     {
575         List<ActivityImpl> actImpls = new ArrayList<ActivityImpl>();
576         Task task = this.getTask(taskId);
577         if (task == null)
578         {
579             throw new ActivitiException("任务不存在");
580         }
581         String processDefinitionId = task.getProcessDefinitionId();
582         String processInstanceId = task.getProcessInstanceId();
583         ProcessDefinitionEntity def =
584             (ProcessDefinitionEntity)((RepositoryServiceImpl)repositoryService)
585             .getDeployedProcessDefinition(processDefinitionId);
586         List<ActivityImpl> activitiList = def.getActivities();// 获得流程定义的所有节点
587         List<String> activeActivityIds =
588             runtimeService.getActiveActivityIds(processInstanceId);// 获取当前活动节点id
589         for (String activeId : activeActivityIds)
590         {
591             for (ActivityImpl activityImpl : activitiList)
592             {
593                 String id = activityImpl.getId();
594                 if (activityImpl.isScope())
595                 {
596                     if (activityImpl.getActivities().size() > 1)
597                     {
598                         List<ActivityImpl> subAcList =
599                             activityImpl.getActivities();
600                         for (ActivityImpl subActImpl : subAcList)
601                         {
602                             String subid = subActImpl.getId();
603                             if (activeId.equals(subid))
604                             {// 获得执行到那个节点
605                                 actImpls.add(subActImpl);
606                                 break;
607                             }
608                         }
609                     }
610                 }
611                 if (activeId.equals(id))
612                 {// 获得执行到那个节点
613                     actImpls.add(activityImpl);
614                 }
615             }
616         }
617         
618         return actImpls;
619     }

701     public List<ActivityImpl> getPreAllTaskAcitiviImpl(
702         List<ActivityImpl> result, String activityId, String processInstanceId)
703     {
704         HistoricProcessInstance hi =
705             this.historyService.createHistoricProcessInstanceQuery()
706                 .processInstanceId(processInstanceId)
707                 .singleResult();
708         ProcessDefinitionEntity processDefinition =
709             (ProcessDefinitionEntity)((RepositoryServiceImpl)repositoryService)
710             .getDeployedProcessDefinition(hi.getProcessDefinitionId());
711         ActivityImpl activityImpl = processDefinition.findActivity(activityId);
712         List<PvmTransition> pvms = activityImpl.getIncomingTransitions();
713         for (PvmTransition pv : pvms)
714         {
715             ActivityImpl temp = (ActivityImpl)pv.getSource();
716             if ("userTask".equals(temp.getProperty("type")))
717             {
718                 result.add(temp);
719                 getPreAllTaskAcitiviImpl(result,
720                     temp.getId(),
721                     processInstanceId);
722             }
723             else if ("startEvent".equals(temp.getProperty("type")))
724             {
725                 return result;
726             }
727             else
728             {
729                 getPreAllTaskAcitiviImpl(result,
730                     temp.getId(),
731                     processInstanceId);
732             }
733         }
734         return null;
735     }

794     /**
795      * 递归获取回退节点
796      * 
797      * @author 张小虎
798      * @param currentActivityImpl 当前的任务对应的活动节点
799      * @param rollBackActivity 最后返回的活动节点集合
800      * @param processInstance 当前流程实例
801      * @throw ActivitiException
802      * @return List<ActivityImpl>
803      */
804     protected List<ActivityImpl> iteratorBackAcitivity(
805         ActivityImpl currentActivityImpl, List<ActivityImpl> rollBackActivity,
806         ProcessInstance processInstance)
807         throws ActivitiException
808     {
809         List<PvmTransition> inComing =
810             currentActivityImpl.getIncomingTransitions();
811         List<PvmTransition> cloneComing = new ArrayList<PvmTransition>();
812         cloneComing.addAll(inComing);
813         this.getUserTaskCommingAndRollBack(currentActivityImpl, 
814             cloneComing, rollBackActivity,processInstance);
815         for (PvmTransition transition : cloneComing)
816         {
817             ActivityImpl activityImpl = (ActivityImpl)transition.getSource();
818             String type = (String)activityImpl.getProperty("type");
819             if ("parallelGateway".equals(type))
820             {
821                 String gatewayId = activityImpl.getId();
822                 if (gatewayId.toUpperCase().startsWith(("FORK")))
823                 {// 并行节点的起点,不能回退，直接返回
824                     return rollBackActivity;
825                 }
826                 this.getGatewayRollBackNode(gatewayId, processInstance,
827                     activityImpl, rollBackActivity);
828             }
829             else if ("userTask".equals(type))
830             {// 任务节点
831                 rollBackActivity.add(activityImpl);
832                 iteratorBackAcitivity(activityImpl,
833                     rollBackActivity,
834                     processInstance);
835             }
836             else if ("exclusiveGateway".equals(type))
837             {// 分支节点
838                 iteratorBackAcitivity(activityImpl,
839                     rollBackActivity,
840                     processInstance);
841             }
842             else if ("startEvent".equals(type))
843             {// 开始节点，不能回退了直接返回
844                 return rollBackActivity;
845             }
846             else
847             {
848                 iteratorBackAcitivity(activityImpl,
849                     rollBackActivity,
850                     processInstance);
851             }
852         }
853         
854         return rollBackActivity;
855     }

857      * 
858     * @Title getUserTaskCommingAndRollBack
859     * @author zhxh
860     * @Description: 获取驳回节点的时候，任务节点的流入和驳回节点
861     * @date 2014-1-6
862     * @param currentActivityImpl
863     * @param cloneComing
864     * @param rollBackActivity
865      */
866     private void getUserTaskCommingAndRollBack(
867         ActivityImpl currentActivityImpl,
868         List<PvmTransition> cloneComing,
869         List<ActivityImpl> rollBackActivity,
870         ProcessInstance processInstance){
871 
872         // 判断是不是任务节点
873         if ("userTask".equals(currentActivityImpl.getProperty("type")))
874         {
875             if (cloneComing.size() > 1)
876             {// 任务前面节点来至于有条件的分支任务
877                 cloneComing.clear();
878                 ActivityImpl lastActivity =
879                     findLastActivity(processInstance, currentActivityImpl);// 找到最近的分支路径
880                 if ("userTask".equals(lastActivity.getProperty("type")))
881                 {
882                     rollBackActivity.add(lastActivity);// 返回的分支第一个节点为任务则添加
883                 }
884                 cloneComing.addAll(lastActivity.getIncomingTransitions());
885             }
886         }
887     }

891     * @author zhxh
892     * @Description: 获取并行节点的驳回节点

894     * @param gatewayId
895     * @param processInstance
896     * @param activityImpl
897     * @param rollBackActivity
898      */
899     private void getGatewayRollBackNode(String gatewayId,ProcessInstance processInstance,
900         ActivityImpl activityImpl,List<ActivityImpl> rollBackActivity)
901     {
902 
903         if (gatewayId.toUpperCase().startsWith(("JOIN")))
904         {// 并行的结束
905          // 并行结束节点
906          // 寻找并行开始节点
907             List<PvmTransition> parallelIn =
908                 activityImpl.getIncomingTransitions();
909             if (parallelIn.size() > 0)
910             {
911                 ActivityImpl returnActivity = null;
912                 List<ActivityImpl> joins =
913                     new ArrayList<ActivityImpl>();
914                 joins.add(activityImpl);
915                 PvmTransition paralleTransation = parallelIn.get(0);
916                 returnActivity =
917                     findParallelGatewayStart(paralleTransation, joins);// 查找join活动节点对应的fork节点
918                 if (returnActivity != null)
919                 {
920                     // 从fork(开始)查询其他节点
921                     iteratorBackAcitivity(returnActivity,
922                         rollBackActivity,
923                         processInstance);
924                 }
925             }
926         }

994     /**
995      * 获取ParallelGateway的join节点对应的fork节点
996      * 
997      * @author 张小虎
998      * @param transition 当前的任务对应的活动节点
999      * @param joins ParallelGateway的join活动节点集合，joins中包含当前join节点
1000      * @throw ActivitiException
1001      * @return ActivityImpl
1002      */
1003     protected ActivityImpl findParallelGatewayStart(PvmTransition transition,
1004         List<ActivityImpl> joins)
1005         throws ActivitiException
1006     {
1007         ActivityImpl activity = (ActivityImpl)transition.getSource();
1008         String type = (String)activity.getProperty("type");
1009         if ("parallelGateway".equals(type))
1010         {
1011             String gatewayId = activity.getId();
1012             if (gatewayId.toUpperCase().startsWith(("FORK")))
1013             {// 并行节点的起点
1014                 joins.remove(0);// 去掉一个join
1015                 if (joins.size() < 1)
1016                 {
1017                     return activity;
1018                 }
1019             }
1020             if (gatewayId.toUpperCase().startsWith(("JOIN")))
1021             {// 并行的结束
1022                 joins.add(activity);
1023                 List<PvmTransition> parallelIn =
1024                     activity.getIncomingTransitions();
1025                 // 选择其中一条路 径
1026                 transition = parallelIn.get(0);
1027                 return findParallelGatewayStart(transition, joins);
1028             }
1029         }
1030         else if ("startEvent".equals(type))
1031         {// 开始节点，不能直接返回
1032             return null;
1033         }
1034         else
1035         {
1036             // 任何节点只选择一条路路径
1037             return findParallelGatewayStart(activity.getIncomingTransitions()
1038                 .get(0), joins);
1039             
1040         }
1041         return null;
1042         
1043     }

1066      *

1074         ActivityImpl activityImpl)
1075         throws ActivitiException
1076     {
1077         List<PvmTransition> inComing = activityImpl.getIncomingTransitions();
1078         ActivityImpl activityImplReturn = null;
1079         Date date = null;
1080         for (PvmTransition transaction : inComing)
1081         {

1083             List<HistoricActivityInstance> historicActivityInstances =
1084                 historyService.createHistoricActivityInstanceQuery()
1085                     .processInstanceId(processInstance.getId())
1086                     .activityId(activity.getId())
1087                     .finished()
1088                     .orderByHistoricActivityInstanceEndTime()
1089                     .desc()
1090                     .list();
1091             if (historicActivityInstances != null
1092                 && historicActivityInstances.size() > 0)
1093             {
1094                 HistoricActivityInstance historicActivityInstance =
1095                     historicActivityInstances.get(0);
1096                 if (date != null)
1097                 {
1098                     if (date.before(historicActivityInstance.getEndTime()))
1099                     {
1100                         date = historicActivityInstance.getEndTime();
1101                         activityImplReturn = activity;
1102                     }
1103                 }
1104                 else
1105                 {
1106                     date = historicActivityInstance.getEndTime();
1107                     activityImplReturn = activity;
1108                 }
1109             }
1110         }
1111         
1112         return activityImplReturn;
1113     }
