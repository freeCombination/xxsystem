113      *

116      * @date 2014-3-4
117      * @param map
118      * @return
119      */
120     @Override
121     public ListVo<ResourceVo> getResourceListRecord(Map<String, String> map)
122         throws BusinessException {
123         Map<String, Object> param = new HashMap<String, Object>();
124         ListVo<ResourceVo> resVoObj = new ListVo<ResourceVo>();
125         List<ResourceVo> resVoList = new ArrayList<ResourceVo>();
126         String resourceName = (String)map.get("resourceName");
127         String currentResId = (String)map.get("currentResId");
128         String parentResId = (String)map.get("parentResId");
129         int start = NumberUtils.toInt(map.get("start"));
130         int limit = NumberUtils.toInt(map.get("limit"));
131         List<Resource> listResult = new ArrayList<Resource>();
132         if (!StringUtils.isBlank(parentResId) && !"0".equals(parentResId)) {
133         	getResListByHierarchicalQuery(Integer.parseInt(parentResId),listResult);
134         }
135         StringBuffer hqBf = new StringBuffer();
136     	for(Resource r : listResult){
137     		hqBf.append(",").append(r.getResourceId());
138     	}
139        
140         StringBuffer sql =
141             new StringBuffer(
142                 "From Resource a where a.status = "
143                     + Constant.STATUS_NOT_DELETE);
144         StringBuffer sqlCount =
145             new StringBuffer(
146                 " select count(*) from Resource a where a.status = "
147                     + Constant.STATUS_NOT_DELETE);
148         
149         if (!StringUtils.isBlank(currentResId)) {
150             sql.append(" and a.resourceId = " + currentResId + "");
151             sqlCount.append(" and a.resourceId = " + currentResId + "");
152         }
153         if (!StringUtils.isBlank(resourceName)) {
154             sql.append(" and a.resourceName like '%" + resourceName + "%'");
155             sqlCount.append(" and a.resourceName like '%" + resourceName + "%'");
156         }
157         if (hqBf.length()>0) {
158             sql.append(" and a.resourceId in( ").append(hqBf.substring(1)).append(")");
159             sqlCount.append(" and a.resourceId in( ").append(hqBf.substring(1)).append(")");
160         }
161         sql.append(" order by a.resourceType.pkDictionaryId,a.resource.resourceId,a.disOrder, a.resourceId ");
162         
163         List<Resource> result = baseDao.query(sql.toString(),null,start,limit);
164       
165         int totalSize =  baseDao.queryTotalCount(sqlCount.toString(), new HashMap<String, Object>());
166             //baseDao.getTotalCount(sql.toString(),null);
167         for (Resource r : result) {
168             ResourceVo rv = new ResourceVo(r);
169             resVoList.add(rv);
170             
171         }
172         
173         resVoObj.setList(resVoList);
174         resVoObj.setTotalSize(totalSize);
175         return resVoObj;
176     }

204     /**
205      * 通过参数获取resource
206      * 
207      * @Title getResourceList
208      * @author wanglc
209      * @date 2013-11-25
210      * @param map 参数Map
211      * @return ListVo<ResourceVo>
212      */
213     @Override
214     public ListVo<ResourceVo> getResourceList(Map<String, String> map)
215         throws BusinessException {
216         Map<String, Object> param = new HashMap<String, Object>();
217         ListVo<ResourceVo> resourceVoObj = new ListVo<ResourceVo>();
218         List<ResourceVo> resourceVoList = new ArrayList<ResourceVo>();
219         String resourceName = (String)map.get("resourceName");
220         String parentResName = (String)map.get("parentResName");
221         String parentResId = (String)map.get("parentResId");
222         int start = NumberUtils.toInt(map.get("start"));
223         int limit = NumberUtils.toInt(map.get("limit"));
224         Resource res = null;
225         StringBuffer hqlQuery =
226             new StringBuffer("select r from Resource r where r.status=0 ");
227         StringBuffer hqlCount =
228             new StringBuffer(
229                 "select count(c.resourceId) from Resource c where c.status=0 ");
230         if (!StringUtils.isBlank(resourceName)) {
231             hqlQuery.append(" and r.resourceName like :resourceName ");
232             hqlCount.append(" and c.resourceName like :resourceName ");
233             param.put("resourceName", "%" + resourceName + "%");
234         }
235         if (!StringUtils.isBlank(parentResName)) {
236             hqlQuery.append(" and r.resource.resourceName = :resourceName ");
237             hqlCount.append(" and c.resource.resourceName = :resourceName ");
238             param.put("parentResName", parentResName);
239         } else if (!StringUtils.isBlank(parentResId)) {
240             if ("0".equals(parentResId)) {
241                 hqlQuery.append(" and r.resource is null ");
242                 hqlCount.append(" and c.resource is null ");
243             } else {
244                 hqlQuery.append(" and r.resource.resourceId = :parentResId");
245                 hqlCount.append(" and c.resource.resourceId = :parentResId");
246                 param.put("parentResId", Integer.parseInt(parentResId));
247             }
248         }
249         hqlQuery.append(" order by r.disOrder,r.resourceId desc ");
250         List<Resource> resourceList =
251             (List<Resource>)this.baseDao.queryEntitysByPage(start,
252                 Integer.MAX_VALUE,
253                 hqlQuery.toString(),
254                 param);
255         int totalSize =
256             this.baseDao.queryTotalCount(hqlCount.toString(), param);
257         
258         for (int i = 0; i < resourceList.size(); i++) {
259             Resource r = resourceList.get(i);
260             ResourceVo rV = new ResourceVo(r);
261             resourceVoList.add(rV);
262         }
263         resourceVoObj.setList(resourceVoList);
264         resourceVoObj.setTotalSize(totalSize);
265         
266         return resourceVoObj;
267     }

466         StringBuffer hql =
467             new StringBuffer("from Resource r where r.status = "

469         Map<String, Object> values = new HashMap<String, Object>();
470         // 遍历Resource中的名字属性和类型属性

875     /**
876      * 验证资源属性的唯一性
877      * 
878      * @Title validateResourceProperties
879      * @author wanglc
880      * @date 2014-2-24
881      * @param paramsMap 参数Map
882      * @return 返回true：验证通过 false：验证不通过
883      */
884     @Override
885     public boolean validateResourceProperties(Map<String, String> paramsMap)
886         throws BusinessException {
887         String key = paramsMap.get("key");
888         String value = paramsMap.get("value");
889         String resourceId = paramsMap.get("resourceId");
890         String validatorType = paramsMap.get("validatorType");
891         String parentId = paramsMap.get("parentId");
892         boolean flag = true;
893         String valueField = "";
894         switch (NumberUtils.toInt(key)) {
895             case 0:
896                 valueField = "resourceName";
897                 break;
898             case 1:
899                 valueField = "code";
900                 break;
901             case 2:
902                 valueField = "urlpath";
903                 break;
904             default:
905                 break;
906         }
907         String hql =
908             "from Resource r where " + " r.resource.resourceId=" + parentId
909                 + "and r." + valueField + " = '" + value + "' and r.status = "
910                 + Constant.STATUS_NOT_DELETE;
911         if (validatorType != null && "update".equals(validatorType)) {
912             hql += " and r.resourceId <> " + resourceId;
913         }
914         int totleSize = baseDao.queryTotalCount(hql, new HashMap());
915         if (totleSize > 0) {
916             flag = false;
917         }
918         return flag;
919     }
