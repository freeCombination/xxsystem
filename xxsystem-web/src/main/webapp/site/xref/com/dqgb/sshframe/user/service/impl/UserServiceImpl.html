266      * 获取用户分页数据
267      * 
268      * @Title getUserPager
269      * @author wanglc
270      * @date 2013-12-6
271      * @param start
272      * @param limit
273      * @param user
274      * @return PagerVo<User>
275      */
276     public PagerVo<User> getUserPager(int start, int limit, User user) {
277         StringBuffer hql = new StringBuffer(" 1=1 ");
278         Map<String, Object> values = new HashMap<String, Object>();
279         // 遍历user中的相关属性
280         if (user != null) {
281             if (StringUtils.isNotBlank(user.getRealname())) {
282                 hql.append(" and obj.realname like ? ");
283                 values.put("realname", "%" + user.getRealname() + "%");
284             }
285             if (StringUtils.isNotBlank(user.getUsername())) {
286                 hql.append(" and obj.username like ? ");
287                 values.put("username", "%" + user.getUsername() + "%");
288             }
289             if (StringUtils.isNotBlank(user.getGender())) {
290                 hql.append(" and obj.gender = ? ");
291                 values.put("gender", user.getGender());
292             }
293             if (user.getEnable() >= 0) {
294                 hql.append(" and obj.enable = ? ");
295                 values.put("enable", user.getEnable());
296             }
297         }
298         hql.append(" and obj.status=" + Constant.STATUS_NOT_DELETE
299             + " order by obj.status,obj.enable,obj.disOrder,obj.userId ");
300         List<User> userList =
301             this.baseDao.queryEntitysByPage(start,
302                 limit,
303                 hql.toString(),
304                 values);
305         int count = baseDao.getTotalCount(hql.toString(), values);
306         return new PagerVo<User>(start, limit, count, userList);
307     }

309     /**
310      * 获取用户列表
311      * 
312      * @Title getUserList
313      * @author wanglc
314      * @date 2013-11-25
315      * @param paramMap 请求参数Map
316      * @return ListVo<UserVo>
317      */
318     @Override
319     public ListVo<UserVo> getUserList(Map<String, String> paramMap)
320         throws BusinessException {
321         int start = NumberUtils.toInt(paramMap.get("start"));
322         int limit = NumberUtils.toInt(paramMap.get("limit"));
323         String orgId = paramMap.get("orgId");
324         String userName = paramMap.get("userName");
325         String flag = paramMap.get("flag");
326         
327         String roleIdOrRoleMemberId = paramMap.get("roleIdOrRoleMemberId");
328         String roleId = paramMap.get("roleId");
329         
330         String orgIds = "";
331         
332         String orgUserHql =
333             "select distinct u from OrgUser ou,User u where u.userId=ou.user.userId and ou.user.isDeletAble = "
334                 + Constant.ALLOW_DELETE + " and ou.isDelete = "
335                 + Constant.STATUS_NOT_DELETE + " and ou.user.status = "
336                 + Constant.STATUS_NOT_DELETE + " and ou.organization.status ="
337                 + Constant.STATUS_NOT_DELETE;
338         
339         if ("QxUser".equals(flag) || StringUtils.isNotBlank(flag)) {
340             orgIds = orgService.getOrgIdsByPermissionScope(orgId, null);
341             if (orgIds != null && !"".equals(orgIds)) {
342                 orgUserHql += " and ou.organization.orgId in (" + orgIds + ")";
343             }
344         }
345         else {
346             if (orgId != null && !"".equals(orgId)) {
347                 orgUserHql += " and ou.organization.orgId in (" + orgId + ")";
348             }
349         }
350         
351         // 当roleIdOrRoleMemberId = “scope”表示数据范围中user添加时限制用户已添加id，此时roleId表示角色成员ID，
352         // 当roleIdOrRoleMemberId != “scope”表示添加角色成员时限制已添加user id，此时roleId表示角色ID
353         if (StringUtils.isNotBlank(roleId)) {
354             if (StringUtils.isNotBlank(roleIdOrRoleMemberId) && "scope".equals(roleIdOrRoleMemberId)) {
355                 // 根据角色成员ID查询该成员已有控制权限下的userIds
356                 List<ScopeMember> userSm = roleService.getUserSm(NumberUtils.toInt(roleId));
357                 if (!CollectionUtils.isEmpty(userSm)) {
358                     String userIds = "";
359                     for (ScopeMember sm : userSm) {
360                         userIds += "," + sm.getUser().getUserId();
361                     }
362                     
363                     orgUserHql += " and u.userId not in (" + userIds.substring(1) + ")";
364                 }
365             }
366             else {
367                 // 根据角色ID查询该角色已有成员的userIds
368                 List<RoleMemberScope> members = roleService.getRoleMemberScopeByRoleId(NumberUtils.toInt(roleId), "userIds");
369                 if (!CollectionUtils.isEmpty(members)) {
370                     String userIds = "";
371                     for (RoleMemberScope m : members) {
372                         userIds += "," + m.getUser().getUserId();
373                     }
374                     
375                     orgUserHql += " and u.userId not in (" + userIds.substring(1) + ")";
376                 }
377             }
378         }
379         
380         if (userName != null && !"".equals(userName)) {
381             orgUserHql += " and ou.user.realname like '%" + userName + "%'";
382         }
383         
384         //int totalSize = baseDao.queryTotalCount(orgUserHql, new HashMap<String, Object>());
385         
386         int beginPos = orgUserHql.toLowerCase().indexOf("from");
387         String hql4Count = " select count(distinct u.userId) " + orgUserHql.substring(beginPos);
388         int totalSize = baseDao.getTotalCount(hql4Count, new HashMap<String, Object>());
389         
390         orgUserHql +=" order by u.disOrder, u.userId";
391         List<User> userList =  (List<User>)baseDao.queryEntitysByPage(start, limit, orgUserHql,new HashMap<String, Object>());
392         
393         ListVo<UserVo> userVoListVo = new ListVo<UserVo>();
394         List<UserVo> userVoList = new ArrayList<UserVo>();
395         for (User u : userList) {
396             UserVo uv = convertVo(u);
397             userVoList.add(uv);
398         }
399         userVoListVo.setList(userVoList);
400         userVoListVo.setTotalSize(totalSize);
401         
402         return userVoListVo;
403     }
