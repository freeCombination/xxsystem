175     public String getResById() {
176         int id = RequestUtil.getInt(getRequest(), "resourceId");
177         if (id != 0) {
178             Resource res;
179             try {
180                 res = resourceService.getResourceById(id);
181                 Map<String, Object> map = new HashMap<String, Object>();
182                 if (res != null) {
183                     map.put("success", true);
184                     map.put("msg", "获取资源成功");
185                     if (res.getResource() != null) {
186                         map.put("parentResourceId", res.getResource()
187                             .getResourceId());
188                         map.put("parentResourceName", res.getResource()
189                             .getResourceName());
190                     }
191                     map.put("resourceId", res.getResourceId());
192                     map.put("resourceTypeId", res.getResourceType()
193                         .getPkDictionaryId());
194                     map.put("dictionaryName", res.getResourceType()
195                         .getDictionaryName());
196                     map.put("resourceName", res.getResourceName());
197                     map.put("code", res.getCode());
198                     map.put("urlpath", res.getUrlpath());
199                     map.put("disOrder", res.getDisOrder());
200                     map.put("remarks", res.getRemarks());
201                 } else {
202                     Resource r = resourceService.getResourceByIdIncludeAll(id);
203                     map.put("success", false);
204                     map.put("msg", "【" + r.getResourceName() + "】已删除，列表已刷新");
205                 }
206                 JsonUtil.outJson(map);
207             } catch (BusinessException e) {
208                 this.excepAndLogHandle(ResourceAction.class,
209                     "通过资源ID获取资源基础数据",
210                     e,
211                     false);
212             }
213         }
214         return null;
215     }

217     /**
218      * 添加资源
219      * 
220      * @Title addResource
221      * @author wanglc
222      * @date 2013-11-25
223      * @return
224      */
225     public String addResource() {
226         try {
227             int resTypeDictId = resource.getResourceType().getPkDictionaryId();// 资源类型
228             String uuid = dictService.getUUIDById(resTypeDictId);
229             
230             if ("".equals(uuid)) {
231                 JsonUtil.outJson("{'success':'false','msg':'选择的资源类型已删除'}");
232             } else {
233                 resource.setResourceTypeDictUUID(uuid);
234                 resource.setCreateDate(new Date());
235                 resource.setStatus(Constant.STATUS_NOT_DELETE);
236                 resource.setRemarks(resource.getRemarks().trim());
237                 //排序如果为空，默认改写为0
238                 if (resource.getDisOrder() == null )
239                 {
240                     resource.setDisOrder(0);
241                 }
242                 resourceService.addResource(resource);
243                 JsonUtil.outJson("{'success':'true','msg':'添加系统资源成功'}");
244                 this.excepAndLogHandle(ResourceAction.class, "添加资源", null, true);
245             }
246         } catch (Exception e) {
247             JsonUtil.outJson("{'success':'false','msg':'添加系统资源失败'}");
248             this.excepAndLogHandle(ResourceAction.class, "添加资源", e, false);
249         }
250         return null;
251     }

355      * @date 2014-3-11
356      * @param resIdsByUser
357      * @param resIdsByRole
358      * @param r
359      * @return
360      */
361     public boolean isChecked(List<Integer> resIdsByUser,
362         List<Integer> resIdsByRole, Resource r) {
363         boolean checkFlag = false;
364         if (resIdsByUser != null && resIdsByUser.contains(r.getResourceId())) {
365             checkFlag = true;
366         }
367         if (resIdsByRole != null && resIdsByRole.contains(r.getResourceId())) {
368             checkFlag = true;
369         }
370         return checkFlag;
371     }

552      * @date 2013-11-25
553      * @param r 资源对象
554      * @return boolean
555      */
556     public boolean decideIsLeaf(Resource r) {
557         boolean result = true;
558         Set<Resource> sonResources = r.getResources();
559         for (Resource sonResource : sonResources) {
560             if (sonResource.getStatus() != Constant.STATUS_IS_DELETED) {
561                 result = false;
562                 break;
563             }
564         }
565         return result;
