546     /**
547      * 获取群组组织机构树
548      * 
549      * @Title getRoleOrgForTree
550      * @author tangh
551      * @date 2014年9月15日
552      * @return String
553      */
554     public String getOrgForTree() {
555         // 获取群组ID
556         String groupId = this.getRequest().getParameter("groupId");
557         Map<String, String> map = new HashMap<String, String>();
558         map.put("groupId", groupId);
559         // 用户存储组织ID
560         List<Integer> orgIds = new ArrayList<Integer>();
561         try {
562             List<GroupMember> gmList = groupMemberService.getDeptByGroupId(map);
563             // 判断群组已经存在的的组织集合是否为空
564             if (gmList != null && gmList.size() > 0) {
565                 // 循环遍历集合，并将组织ID封装到集合中
566                 for (GroupMember groupMember : gmList) {
567                     orgIds.add(groupMember.getOrg().getOrgId());
568                 }
569             }
570             // 获取当前组织的一级菜单
571             List<Organization> res = groupMemberService.getFirstLevelOrg();
572             // 封装组织机构树
573             buffer.append("[");
574             for (Organization r : res) {
575                 buffer.append("{\"nodeId\":\"" + r.getOrgId() + "\",");
576                 buffer.append("\"checked\":" + orgIds.contains(r.getOrgId())
577                     + ",");
578                 buffer.append("\"leaf\":" + decideIsLeafOrg(r) + ",");
579                 buffer.append("\"text\":\"" + r.getOrgName() + "\",");
580                 buffer.append("\"children\":[");
581                 getByParentOrg(r, orgIds);
582                 buffer.append("]},");
583             }
584             buffer.append("]");
585             String result = buffer.toString().replaceAll(",]", "]");
586             JsonUtil.outJsonArray(result);
587         }
588         catch (Exception e) {
589             this.excepAndLogHandle(GroupManagerAction.class,
590                 "获取群组组织机构树",
591                 e,
592                 false);
593         }
594         return null;
595     }

597     /**
598      * 根据父节点查询所有子节点
599      * 
600      * @Title getByParentOrg
601      * @author tangh
602      * @date 2014年9月18日
603      * @param org 组织对象
604      * @param orgIds 组织
605      * @return String
606      */
607     private void getByParentOrg(Organization org, List<Integer> orgIds) {
608         try {
609             List<Organization> list =
610                 groupMemberService.getAllSonOrgByParentId(org.getOrgId() + "");
611             // 判断集合是否为空
612             if (list.size() != 0) {
613                 // 循环遍历集合并封装数据
614                 for (Organization r : list) {
615                     buffer.append("{\"nodeId\":\"" + r.getOrgId() + "\",");
616                     buffer.append("\"text\":\"" + r.getOrgName() + "\",");
617                     buffer.append("\"checked\":"
618                         + orgIds.contains(r.getOrgId()) + ",");
619                     buffer.append("\"leaf\":" + decideIsLeafOrg(r) + ",");
620                     buffer.append("\"children\":[");
621                     getByParentOrg(r, orgIds);
622                     buffer.append("]},");
623                 }
624             }
625         }
626         catch (Exception e) {
627             this.excepAndLogHandle(GroupManagerAction.class,
628                 "根据父节点查询所有子节点",
629                 e,
630                 false);
631         }
632     }

634     /**
635      * 判断树节点是否为叶子节点
636      * 
637      * @Title decideIsLeafOrg
638      * @author tangh
639      * @date 2014年9月19日
640      * @param org 组织对象
641      * @return
642      */
643     public boolean decideIsLeafOrg(Organization org) {
644         boolean result = true;
645         Set<Organization> sonOrganization = org.getOrganizations();
646         for (Organization sonOrg : sonOrganization) {
647             if (sonOrg.getStatus() != Constant.STATUS_IS_DELETED) {
648                 result = false;
649                 break;
650             }
651         }
652         return result;
653     }
