38  @Entity
39  @Table(name = "T_WORK_TEAM")
40  public class WorkTeam implements Serializable {
41      
42      /**
43       * @Fields serialVersionUID : 对象序列化ID
44       */
45      private static final long serialVersionUID = 1230051032089053161L;
46      
47      /**
48       * @Fields id : 班组ID

50      private Integer id;
51      
52      /**

57      /**
58       * @Fields monitor : 班组负责人，或叫班长

60      private User monitor;

62      /**
63       * @Fields teamName : 班组名称

65      private String teamName;

67      /**
68       * @Fields createDate : 创建时间

82       * @Title getId

85       * @date 2014-8-22
86       * @return 主键
87       */
88      @Id
89      @GeneratedValue(strategy = GenerationType.AUTO)

95      /**

105      * @Description: 班组所属组织
106      * @date 2014-8-22
107      * @return 班组所属组织
108      */
109     @ManyToOne(fetch = FetchType.LAZY)
110     @JoinColumn(name = "ORG_ID", nullable = false)

112         return this.org;
113     }
114     
115     /**
116      * @param org the org to set
117      */
118     public void setOrg(Organization org) {
119         this.org = org;
120     }
121

123      * @Title getMonitor
124      * @author dong.he
125      * @Description: 班组负责人，或叫班长
126      * @date 2014-8-22
127      * @return 班组负责人，或叫班长
128      */
129     @ManyToOne(fetch = FetchType.LAZY)
130     @JoinColumn(name = "MONITOR_ID", nullable = true)

132         return monitor;

135     /**
136      * @param monitor the monitor to set
137      */
138     public void setMonitor(User monitor) {
139         this.monitor = monitor;
140     }

146      * @date 2014-8-22

149     @Column(name = "TEAM_NAME", nullable = false, length = 100)

153     
154     /**
155      * @param teamName the teamName to set
156      */
157     public void setTeamName(String teamName) {
158         this.teamName = teamName;
159     }

174     /**
175      * @param createDate the createDate to set
176      */
177     public void setCreateDate(Date createDate) {
178         this.createDate = createDate;
179     }

192     
193     /**
194      * @param remark the remark to set
195      */
196     public void setRemark(String remark) {
