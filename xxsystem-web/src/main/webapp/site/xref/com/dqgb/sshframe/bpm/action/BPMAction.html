194         {
195             String usersId = this.getRequest().getParameter("usersId");
196             String userName = this.getRequest().getParameter("userName");
197             String start = getRequest().getParameter("start");
198             String limit = getRequest().getParameter("limit");

200             paramMap.put("usersId", usersId);

207             List<Map> userVoList = new ArrayList<Map>();
208             for (User user : userList)

220             userVoListVo.setList(userVoList);

231     /**
232      * 
233     * @Title collectionUnitName
234     * @author zhxh
235     * @Description: 针对BPMAction收集单位名称
236     * @see BPMAction#getUserListForFlow
237     * @date 2014-1-16
238     * @param orgsUsers
239     * @return String
240      */
241     private String collectionUnitName(Set<OrgUser> orgsUsers)
242     {
243        StringBuffer unitNames =new StringBuffer();
244        String unitName = null;
245         if (!CollectionUtils.isEmpty(orgsUsers))
246         {
247             for (OrgUser orgUser : orgsUsers)
248             {
249                 if (orgUser.getIsDelete() == Constant.STATUS_NOT_DELETE
250                     && orgUser.getOrganization() != null
251                     && orgUser.getOrganization().getEnable() == Constant.ENABLE
252                     && orgUser.getOrganization().getStatus() == Constant.STATUS_NOT_DELETE)
253                 {
254                     unitNames.append(orgUser.getOrganization()
255                         .getOrgName());
256                 }
257                 if (unitNames.length() > 0)
258                 {
259                     unitName = unitNames.substring(0, unitNames.length());
260                 }
261             }
262             
263         }
264         return unitName;
265     }

483                     map.put("loginName", u.getUsername());

486                     
487                 }

489                 {
490                     for (String str : strs)
491                     {
492                         if (mapTemp.get(str.trim()) == null)
493                         {
494                             Map map = new HashMap();
495                             map.put("loginName", str);
496                             map.put("userName", "");
