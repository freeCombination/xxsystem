47       * 得到指定路径下的xml的document
48       * 
49       * @Title getDocument
50       * @author wanglc
51       * @Description:
52       * @date 2013-12-11
53       * @param xmlPath
54       * @return Document
55       */
56      public static Document getDocument(String xmlPath) {
57          Document document = null;
58          try {
59              document =
60                  new SAXReader().read(Thread.currentThread()
61                      .getContextClassLoader()
62                      .getResourceAsStream(xmlPath));
63          }
64          catch (Exception e) {
65              
66          }
67          return document;
68      }

70      /**
71       * 只将叶子节点中的元素，放入map中，数据无重复
72       * 
73       * @Title getXMLInfo
74       * @author 张小虎
75       * @Description:
76       * @date 2013-12-11
77       * @param element
78       * @param ret
79       * @return Map<String,Object>
80       */
81      public static Map<String, Object> getXMLInfo(Element element,
82          Map<String, Object> ret) {
83          if (element.elements().size() < 1) {
84              ret.put(element.getQName().getName(), element.getText());
85              return ret;
86          }
87          else {
88              for (Iterator it = element.elementIterator(); it.hasNext();) {
89                  getXMLInfo((Element)it.next(), ret);
90              }
91              
92          }
93          return null;
94          
95      }
96      
97      /**
98       * 将叶子节点中的元素，放入map中，并存入list
99       * 
100      * @Title getXMLInfo
101      * @author 张小虎
102      * @Description:
103      * @date 2013-12-11
104      * @param fileUrl
105      * @param xpath
106      * @return List<Map<String,Object>>
107      * @throws Exception
108      */
109     public static List<Map<String, Object>> getXMLInfo(String fileUrl,
110         String xpath)
111         throws Exception {
112         Document document =
113             new SAXReader().read(new FileInputStream(new File(fileUrl)));
114         List<Map<String, Object>> ret = new ArrayList<Map<String, Object>>();
115         List<Element> dataItemList = (List<Element>)document.selectNodes(xpath);
116         for (Element e : dataItemList) {
117             Map tempMap = new HashMap();
118             getXMLInfo(e, tempMap);
119             ret.add(tempMap);
120         }
121         return ret;
122         
123     }

125     /**
126      * 获取xml中的元素
127      * 
128      * @Title getElementList
129      * @author 张小虎
130      * @Description:
131      * @date 2013-12-11
132      * @param doc
133      * @param resultList 返回元素list
134      * @param fileAbsoluteUrl 文件的绝对路径
135      * @param elementName 查找的元素名称
136      * @return List<Element>
137      * @throws Exception
138      */
139     public static List<Element> getElementList(Document doc,
140         List<Element> resultList, String fileAbsoluteUrl, String elementName)
141         throws Exception {
142         try {
143             Document document = doc;
144             if (doc == null) {
145                 InputStream in = new FileInputStream(new File(fileAbsoluteUrl));
146                 document = new SAXReader().read(in);
147             }
148             Element root = document.getRootElement();
149             if (root.getQName().getName().trim().equals(elementName)) {
150                 resultList.add(root);
151             }
152             getElementList(resultList, root, elementName);
153         }
154         catch (FileNotFoundException e) {
155             throw new Exception(fileAbsoluteUrl + ",文件未找到！");
156         }
157         return resultList;
158     }

160     /**
161      * 递归查找xml中的元素
162      * 
163      * @Title getElementList
164      * @author 张小虎
165      * @Description:
166      * @date 2013-12-11
167      * @param resultList 返回元素list
168      * @param element
169      * @param elementName 查找的元素名称
170      * @return
171      */
172     public static List<Element> getElementList(List<Element> resultList,
173         Element element, String elementName) {
174         
175         if (element.elements().size() < 1) {
176             return resultList;
177         }
178         else {
179             for (Iterator it = element.elementIterator(); it.hasNext();) {
180                 Element e = (Element)it.next();
181                 if (e.getQName().getName().trim().equals(elementName)) {
182                     resultList.add(e);
183                 }
184                 getElementList(resultList, e, elementName);
185             }
186             
187         }
188         return null;
189         
190     }

192     /**
193      * 修改系统维护中的数据
194      * 
195      * @Title writeXml
196      * @author 张小虎
197      * @Description:
198      * @date 2013-12-11
199      * @param document
200      * @param targetFileUrl
201      * @throws IOException
202      */
203     public static void writeXml(Document document, String targetFileUrl)
204         throws IOException {
205         File file = new File(targetFileUrl);
206         OutputFormat format = OutputFormat.createPrettyPrint();
207         XMLWriter writer = new XMLWriter(new FileOutputStream(file), format);
208         writer.write(document);
209         writer.close();
210     }

212     /**
213      * 获取元素 document 不是null则按照document，否则按照inputStream,再次为reativeXmlPath
214      * 
215      * @Title getElement
216      * @author 张小虎
217      * @Description:
218      * @date 2013-12-11
219      * @param document
220      * @param inputStream
221      * @param reativeXmlPath
222      * @param xPath
223      * @return
224      * @throws DocumentException
225      */
226     public static List<Element> getElement(Document document,
227         InputStream inputStream, String reativeXmlPath, String xPath)
228         throws DocumentException {
229         List<Element> selects = new ArrayList();
230         if (document != null) {
231             selects = document.selectNodes(xPath);
232         }
233         else if (inputStream != null) {
234             document = new SAXReader().read(inputStream);
235             selects = document.selectNodes(xPath);
236         }
237         else if (StringUtils.isNotBlank(reativeXmlPath)) {
238             document = getDocument(reativeXmlPath);
239             selects = document.selectNodes(xPath);
240         }
241         return selects;
242     }
