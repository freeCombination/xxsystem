243     /**
244      * 查询群组
245      * 
246      * @Title getGroup
247      * @author tangh
248      * @date 2014年9月15日
249      * @param paramMap
250      * @return ListVo<GroupVo> 群组集合
251      * @throws BusinessException
252      */
253     @Override
254     public ListVo<GroupVo> getGroup(Map<String, String> paramMap)
255         throws BusinessException {
256         ListVo<GroupVo> listVo = new ListVo<GroupVo>();
257         List<Group> list = new ArrayList<Group>();
258         int totalSize = 0;
259         
260         // 获取参数
261         String start = paramMap.get("start");
262         String limit = paramMap.get("limit");
263         String groupName = paramMap.get("groupName");
264         String flag = paramMap.get("flag");
265         String roleId = paramMap.get("roleId");
266         StringBuffer buffer = new StringBuffer();
267         if ("QxGroup".equals(flag) || StringUtils.isNotBlank(flag)) {
268             
269             StringBuffer hql1 = new StringBuffer();
270             hql1.append(" from RoleMemberScope rms where 1=1 and rms.group.id is not null  ");
271             hql1.append(" and rms.user.userId is null ");
272             hql1.append(" and rms.role.roleId = " + roleId + " ");
273             
274             List<RoleMemberScope> gList = baseDao.queryEntitys(hql1.toString());
275             
276             // 判断gList是否为空
277             if (gList != null && gList.size() > 0) {
278                 // 遍历群组list集合
279                 for (int i = 0; i < gList.size(); i++) {
280                     // 判断是否为最后一个元素，不是就添加，分割
281                     if (i != gList.size() - 1) {
282                         buffer.append(gList.get(i).getGroup().getId() + ",");
283                     }
284                     else {
285                         buffer.append(gList.get(i).getGroup().getId());
286                     }
287                 }
288             }
289         }
290         // 查询hql语句
291         StringBuffer hql = new StringBuffer();
292         hql.append("from Group o where 1=1 ");
293         
294         // 判断buffer是否为空，筛选群组的id
295         if (StringUtils.isNotBlank(buffer.toString())) {
296             hql.append(" and o.id not in (" + buffer.toString() + ") ");
297         }
298         
299         // 判断班组名称是否为空
300         if (StringUtils.isNotBlank(groupName)) {
301             hql.append(" and o.groupName like '%" + groupName + "%' ");
302         }
303         
304         hql.append(" order by o.id");
305         // 统计中条数
306         totalSize =
307             baseDao.queryTotalCount(hql.toString(),
308                 new HashMap<String, Object>());
309         try {
310             list =
311                 (List<Group>)baseDao.queryEntitysByPage(Integer.parseInt(start),
312                     Integer.parseInt(limit),
313                     hql.toString(),
314                     new HashMap<String, Object>());
315             // 用于存储群组VO的集合
316             List<GroupVo> groupVoList = new ArrayList<GroupVo>();
317             GroupVo groupVo = null;
318             // 循环遍历结果集，封装对象并添加群组VO集合
319             for (Group group : list) {
320                 // 封装对象
321                 groupVo = new GroupVo();
322                 groupVo.setId(group.getId());
323                 groupVo.setGroupName(group.getGroupName());
324                 groupVo.setRemark(group.getRemark());
325                 groupVoList.add(groupVo);
326             }
327             listVo.setList(groupVoList);
328             listVo.setTotalSize(totalSize);
329         }
330         catch (Exception e) {
331             throw new BusinessException(e.getMessage());
332         }
333         return listVo;
334     }

336     /**
337      * 验证群组名字
338      * 
339      * @Title validateGroupName
340      * @author tangh
341      * @date 2014年9月15日
342      * @param paramMap 参数
343      * @return Integer 整形
344      * @throws BusinessException
345      */
346     @Override
347     public Map<String, Object> validateGroupName(Map<String, String> paramMap)
348         throws BusinessException {
349         Map<String, Object> validate = new HashMap<String, Object>();
350         // 获取参数
351         String groupName = paramMap.get("groupName");
352         int totleSize = 0;
353         
354         // 判断参数是否为空
355         if (StringUtils.isNotBlank(groupName)) {
356             // 根据班组名称判断该班组名称是否已经使用
357             StringBuffer hql = new StringBuffer();
358             hql.append("from Group o where 1=1 ");
359             hql.append(" and o.groupName='" + groupName + "'");
360 
361             // 总条数
362             totleSize =
363                 baseDao.queryTotalCount(hql.toString(),
364                     new HashMap<String, Object>());
365         }
366         
367         if (totleSize > 0) {
368             validate.put("success", true);
369             validate.put("valid", false);
370             validate.put("reason", "数据已存在");
371         }
372         else {
373             validate.put("success", true);
374             validate.put("valid", true);
375             validate.put("reason", "");
376         }
377         return validate;
378     }

385      * @Description:
386      * @date 2014年9月15日
387      * @param memberId
388      * @return
389      */
390     @Override
391     public List<GroupMember> getGroupByMemberId(String memberIds, String type)
392         throws BusinessException {
393         List<GroupMember> groupMembers = new ArrayList<GroupMember>();
394         String hql = "from GroupMember gm where 1=1";
395         // memberId不为空
396         if (!"".equals(memberIds) && memberIds.length() != 0) {
397             if (type.equals(Constant.ORG)) {
398                 hql += "and  gm.org.orgId in (" + memberIds + ")";
399             }
400             else if (type.equals(Constant.USER)) {
401                 hql += "and  gm.user.userId in (" + memberIds + ")";
402             }
403             groupMembers = baseDao.queryEntitys(hql);
404         }
405         return groupMembers;
406     }
