175         if (mapList != null && mapList.size() > 0) {

333     /**
334      * 
335      * @Title findPageBySql
336      * @author zhxh
337      * @Description:纯sql分页查询
338      * @date 2013-12-28
339      * @return
340      */
341     public List<Object[]> findPageBySql(final String sql, final int start,
342         final int limit) {
343         return this.getHibernateTemplate().executeFind(new HibernateCallback() {
344             public Object doInHibernate(Session session)
345                 throws HibernateException, SQLException {
346                 Query query = session.createSQLQuery(sql);
347                 query.setFirstResult(start);
348                 query.setMaxResults(limit);
349                 return query.list();
350             }
351         });
352     }

354     /**
355      * 
356      * @Title getRoleTaskSql
357      * @author zhxh
358      * @Description:任务是角色的sql
359      * @date 2013-12-30
360      * @param u 用户
361      * @param sqlSelect 选择的sql
362      * @param sqlConditionWhere 用户查询条件sql
363      * @return String
364      */
365     private String getRoleTaskSql(User u, StringBuffer sqlSelect,
366         StringBuffer sqlConditionWhere) {
367         StringBuffer sqlRole = new StringBuffer();
368         sqlRole.append(" union ");
369         sqlRole.append(sqlSelect);
370         sqlRole.append(" from ACT_RU_TASK a,SYS_ACTIVITI_PROCESS_INST b,SYS_ACTIVITI_CATEGORY c");
371         sqlRole.append(" where a.ASSIGNEE_ is null");
372         sqlRole.append(" and b.FK_ACTIVITI_CATEGORY_ID=c.PK_ACTIVITI_CATEGORY_ID");
373         sqlRole.append(" and b.PROCESS_INSTANCE_ID=a.PROC_INST_ID_");
374         sqlRole.append(sqlConditionWhere);
375         sqlRole.append(" and exists(select 1 from t_ROLE e,t_role_member_scope f,t_USER g,"
376             + "  SYS_ACTIVITI_ROLE h,t_ORGANIZATION i");
377         sqlRole.append(" where e.ROLE_ID=f.ROLE_ID  ");
378         sqlRole.append(" and h.TASKID=a.ID_");
379         sqlRole.append(" and e.ROLE_ID=h.FK_ROLE_ID");
380         sqlRole.append(" and f.USER_ID=g.USER_ID");
381         sqlRole.append(" and f.ORG_ID=i.ORG_ID");
382         sqlRole.append(" and ( (INSTR(c.category_code,'_')>0 ");
383         sqlRole.append(" and  i.ORG_CODE=substr(c.category_code,INSTR(c.category_code,'_')+1 ) ) "
384             + " or (INSTR(c.category_code,'_')<=0)) ");
385         sqlRole.append(" and g.USERNAME='" + u.getUsername()).append("' ");
386         sqlRole.append(" and g.STATUS=" + Constant.STATUS_NOT_DELETE)
387             .append(")");
388         return sqlRole.toString();
389     }
