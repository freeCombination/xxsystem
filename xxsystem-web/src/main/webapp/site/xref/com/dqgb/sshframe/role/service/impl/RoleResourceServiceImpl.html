34  @SuppressWarnings("unchecked")
35  @Service("roleResourceService")
36  public class RoleResourceServiceImpl implements IRoleResourceService {
37      
38      @Autowired
39      @Qualifier("baseDao")
40      private IBaseDao baseDao;
41      
42      public void setBaseDao(IBaseDao baseDao) {
43          this.baseDao = baseDao;
44      }
45      
46      /**
47       * 获取角色资源
48       * 
49       * @Title getOwnResByUserRole
50       * @author wanglc
51       * @date 2013-11-25
52       * @param roleIds 角色ID数组字符串，以逗号分隔
53       * @return
54       */
55      @Override
56      public List<RoleResource> getOwnResByUserRole(String roleIds)
57          throws BusinessException {
58          if (roleIds != null && !"".equals(roleIds)) {
59              String hql =
60                  "from RoleResource rr where rr.role.roleId in (" + roleIds
61                      + ")";
62              List<RoleResource> rrList = baseDao.queryEntitys(hql);
63              if (rrList.size() > 0) {
64                  return rrList;
65              }
66          }
67          return null;
68      }

70  }
