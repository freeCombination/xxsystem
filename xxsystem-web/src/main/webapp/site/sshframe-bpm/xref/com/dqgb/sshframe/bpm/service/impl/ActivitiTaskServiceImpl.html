315      * @Description:获取任务的种类数据

317      */
318     private void getCategoryData(List<TaskVo> resultList, String categoryIds) {
319         if (categoryIds.length() > 0) {
320             List<ActivitiCategory> categories =
321                 activitiCategoryService.getActivitiCategoryById(categoryIds);
322             for (ActivitiCategory category : categories) {
323                 for (TaskVo taskVo : resultList) {
324                     if (taskVo.getCategoryId() == category.getCategoryId()
325                         .intValue() && category.getParentId() != 0) {
326                         taskVo.setCategoryCode(category.getCode());
327                         taskVo.setCategoryName(category.getName());
328                         ActivitiCategory top =
329                             this.activitiCategoryService.getActivitiTopCategory(category.getCategoryId());
330                         if (top != null) {
331                             taskVo.setTopCategoryCode(top.getCode());
332                             taskVo.setTopCategoryId(top.getCategoryId());
333                             taskVo.setTopCategoryName(top.getName());
334                         }
335                     } else if (taskVo.getCategoryId() == category.getCategoryId()
336                         .intValue()
337                         && category.getParentId() == 0) {
338                         taskVo.setCategoryCode(category.getCode());
339                         taskVo.setCategoryName(category.getName());
340                         taskVo.setTopCategoryCode(category.getCode());
341                         taskVo.setTopCategoryId(category.getCategoryId());
342                         taskVo.setTopCategoryName(category.getName());
343                     }
344                 }
345             }
346         }

391     @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
392     @Override
393     public void rejectTaskForApply(Map paramMap, Map commitMap) {
394         ActivitiProcessApproval processApproval = null;
395         InputStream in = null;
396         try {
397             User u = (User)paramMap.get("user");
398             Task task =
399                 this.activitiAndBusinessService.getTask((String)paramMap.get("taskId"));
400             if (task == null) {
401                 throw new ServiceException("任务不存在!");
402             }
403             in =
404                 this.activitiAndBusinessService.getActivitiXmlByProcessDefineId(task.getProcessDefinitionId());
405             List<Element> resultList = new ArrayList<Element>();
406             Document doc = new SAXReader().read(in);
407             resultList =
408                 XMLUtil.getElementList(doc, resultList, null, "userTask");
409             String taskDefineKey = getBpmApplyTaskDefine(resultList);
410             List<ActivityImpl> activityImpls =
411                 this.activitiAndBusinessService.getRollBackActivity((String)paramMap.get("taskId"));
412             ActivityImpl a =
413                 this.getBpmApplyActivityImpl(activityImpls, taskDefineKey);
414             if (a == null) {
415                 throw new ServiceException("回滚之前没有任何申请节点");
416             }
417             processApproval =
418                 this.activitiAndBusinessService.generateActivitiProcessApproval(paramMap,
419                     task,
420                     0);
421             this.activitiAndBusinessService.addActivitiProcessApproval(processApproval);
422             this.generateFormUrl(paramMap, task);
423             if (StringUtils.isBlank(task.getAssignee())) {
424                 task.setAssignee(u.getUsername());
425             }
426             this.activitiAndBusinessService.saveTask(task);
427             this.activitiAndBusinessService.rollbackTask(task.getId(),
428                 a.getId(),
429                 commitMap);
430             List<ActivitiProcessInstanceVo> activitiProcessInstanceVos =
431                 this.activitiAndBusinessService.getActivitiProcessInstance(task.getProcessInstanceId());
432             if (CollectionUtils.isEmpty(activitiProcessInstanceVos)) {
433                 throw new ServiceException("流程实例发起人为空!");
434             } else {
435                 setTaskDelegate(task.getTaskDefinitionKey(),
436                     task.getProcessInstanceId(),
437                     activitiProcessInstanceVos.get(0).getUserId(),
438                     SystemConstant.PROCESS_AUTO_LOGINNAME);
439             }
440             
441         } catch (Exception e) {
442             throw new ServiceException(e.getMessage(), e);
443         } finally {
444             this.closeInputStream(in);
445         }
446     }

473      * @Description:获取申请节点的任务定义key
474      * @date 2014-1-6
475      * @param resultList
476      * @return String
477      */
478     private String getBpmApplyTaskDefine(List<Element> resultList) {
479         String taskDefineKey = "";
480         for (Element e : resultList) {
481             if ("bpm_apply".equals(e.attributeValue("taskType"))) {
482                 taskDefineKey = e.attributeValue("id");
483                 break;
484             }
485         }
486         if ("".equals(taskDefineKey)) {
487             throw new ServiceException("没有申请节点");
488         }
489         return taskDefineKey;
490     }

547         TaskVo vo = null;

549             vo = new TaskVo();

552             vo.setName(h.getName());
553             vo.setOwner(h.getOwner());

555             vo.setTaskDefinitionKey(h.getTaskDefinitionKey());
556             vo.setProcessInstanceId(h.getProcessInstanceId());
557             vo.setAssignee(h.getAssignee());
558             if (h.getEndTime() != null) {
559                 vo.setEndTime(DateUtil.dateToString(h.getEndTime(),
560                     SystemConstant.DATE_PATTEN_LONG));
561                 if (h.getDeleteReason() != null
562                     && !"completed".equals(h.getDeleteReason())) {
563                     vo.setTaskStatus("2");
564                 } else {
565                     vo.setTaskStatus("1");
566                 }
567             }
568             
569             if (h.getStartTime() != null) {
570                 vo.setCreateTime(DateUtil.dateToString(h.getStartTime(),
571                     SystemConstant.DATE_PATTEN_LONG));
572             }
573             if (h.getDurationInMillis() != null) {
574                 vo.setDurationTime(this.activitiAndBusinessService.convertTimeToChinese(h.getDurationInMillis()));
575             }
576             List<ActivitiProcessInstanceVo> activitiProcessInstanceVos =
577                 this.activitiAndBusinessService.getActivitiProcessInstance(vo.getProcessInstanceId());
578             if (activitiProcessInstanceVos.size() > 0) {
579                 vo.setCategoryId(activitiProcessInstanceVos.get(0)
580                     .getCategoryId());
581                 vo.setCategoryCode(activitiProcessInstanceVos.get(0)
582                     .getCategoryCode());
583                 vo.setCategoryName(activitiProcessInstanceVos.get(0)
584                     .getCategoryName());
585                 ActivitiCategory top =
586                     this.activitiCategoryService.getActivitiTopCategory(activitiProcessInstanceVos.get(0)
587                         .getCategoryId());
588                 if (top != null) {
589                     vo.setTopCategoryCode(top.getCode());
590                     vo.setTopCategoryId(top.getCategoryId());
591                     vo.setTopCategoryName(top.getName());
592                 }
593             }
594         }
595         
596         return vo;
597     }

600     @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
601     public Map<String,String> completeTask(Map paramMap, Map commitMap)
602         throws ServiceException {
603         InputStream in = null;
604         Map<String,String> result =null;
605         try {
606             ActivitiProcessApproval processApproval = new ActivitiProcessApproval();
607             // 是否为审批节点
608             Task task = this.activitiAndBusinessService.getTask((String)paramMap.get("taskId"));
609             
610             if (task == null) {
611                 throw new ServiceException("任务不存在");
612             }
613             in = this.activitiAndBusinessService.getProcessXmlResource(task.getProcessInstanceId());
614            
615             result =  this.checkExaapproveNode(in, task,  (String)paramMap.get("pass"), processApproval);
616             task.setDescription((String)paramMap.get("opinion"));
617             processApproval = this.activitiAndBusinessService.generateActivitiProcessApproval(paramMap,task,2);
618             
619             Object userObj = paramMap.get("user");
620             if (task == null) {
621                 throw new ServiceException("任务不存在");
622             }
623             // 人员为空设置人员
624             if (StringUtils.isBlank(task.getAssignee()) && userObj !=null) {
625                 task.setAssignee(((User)userObj).getUsername());
626             }
627             
628             this.generateFormUrl(paramMap, task);
629             this.activitiAndBusinessService.saveTask(task);
630             this.checkAssigneeUserMap(commitMap);
631             this.activitiAndBusinessService.completeTask(task.getId(), commitMap);
632             this.activitiAndBusinessService.addActivitiProcessApproval(processApproval);
633             
634             if("false".equals((String)paramMap.get("pass"))){
635          	   result.put("finalNode", "true");
636             }else{
637 	            HistoricProcessInstance pi = activitiAndBusinessService.getHistoricProcessInstance(task.getProcessInstanceId());
638 	            if(pi !=null && pi.getEndTime() !=null){

750         throws ServiceException {
751         InputStream in = null;
752         try {
753             Task task =
754                 this.activitiAndBusinessService.getTask((String)paramMap.get("taskId"));
755             User u = (User)paramMap.get("user");
756             if (task == null) {
757                 throw new ServiceException("任务不存在!");
758             }
759             if (commitMap == null) {
760                 commitMap = new HashMap();
761             }
762             this.checkRejectAssigneeUser(task, commitMap, rejeckToTaskDefineKey);
763             ActivitiProcessApproval processApproval =
764                 this.activitiAndBusinessService.generateActivitiProcessApproval(paramMap,
765                     task,
766                     0);

769             if (StringUtils.isBlank(task.getAssignee())) {
770                 task.setAssignee(u.getUsername());
771             }
772             this.activitiAndBusinessService.saveTask(task);
773             this.activitiAndBusinessService.rollbackTask(task.getId(),
774                 rejeckToTaskDefineKey,
775                 commitMap);
776             
777             this.activitiAndBusinessService.addActivitiProcessApproval(processApproval);
778             // 判断rejeckToTaskDefineKey是不是申请
779             in =
780                 this.activitiAndBusinessService.getActivitiXmlByProcessDefineId(task.getProcessDefinitionId());
781             List<Element> resultList = new ArrayList<Element>();
782             Document doc = new SAXReader().read(in);
783             resultList =
784                 XMLUtil.getElementList(doc, resultList, null, "userTask");
785             
786             String taskDefineKey = getBpmApplyTaskDefine(resultList);
787             if (rejeckToTaskDefineKey.equals(taskDefineKey)) {
788                 List<ActivitiProcessInstanceVo> activitiProcessInstanceVos =
789                     this.activitiAndBusinessService.getActivitiProcessInstance(task.getProcessInstanceId());
790                 if (CollectionUtils.isEmpty(activitiProcessInstanceVos)) {
791                     throw new ServiceException("流程实例发起人为空!");
792                 } else {
793                     setTaskDelegate(rejeckToTaskDefineKey,
794                         task.getProcessInstanceId(),
795                         activitiProcessInstanceVos.get(0).getUserId(),
796                         SystemConstant.PROCESS_AUTO_LOGINNAME);
797                 }
798             }
799         } catch (Exception e) {
800             throw new ServiceException(e.getMessage(), e);
801         }

877         paramMap.remove("user");

894                     .append("=")

896             }
897             task.setFormURL(task.getFormURL() + otherParam);
898         }
