55      @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
56      @Override
57      public void addActivitiCategory(ActivitiCategory category)
58          throws ServiceException
59      {
60          try
61          {
62              Map paramMap = new HashMap();
63              paramMap.put("code", category.getCode());
64              paramMap.put("isDelete", 0);
65              List list = this.getActivitiCategory(paramMap);
66              if (list.size() > 0)
67              {
68                  throw new ServiceException("类型编码已经存在！");
69              }
70              if (category.getParentId() != 0)
71              {
72                  paramMap.clear();
73                  paramMap.put("categoryId", category.getParentId());
74                  paramMap.put("isDelete", 0);
75                  list = this.getActivitiCategory(paramMap);
76                  if (list.size() < 1)
77                  {
78                      throw new ServiceException("父级类型已经不存在！");
79                  }
80              }
81              this.activitiCategoryDao.addActivitiCategory(category);
82              this.createBpmFolder(category);
83          }
84          catch (ServiceException e)
85          {
86              throw new ServiceException(e.getMessageKey(),e);
87          }
88          catch (Exception e)
89          {
90              throw new ServiceException("保存失败!",e);
91          }
92          
93      }

128     @Transactional(readOnly = false, propagation = Propagation.REQUIRED)
129     @Override
130     public void updateActivitiCategory(ActivitiCategory category)
131         throws ServiceException
132     {
133         try
134         {
135             List list =
136                 this.activitiCategoryDao.queryEntitys(
137                     "from ActivitiCategory " +
138                 		" where id=? and isDelete=0",
139                     new Object[] {category.getCategoryId()});
140             if (list.size() < 1)
141             {
142                 throw new ServiceException("数据不存在");
143             }
144             ActivitiCategory tempCategory = (ActivitiCategory)list.get(0);
145             
146             if (!category.getCode().equals(tempCategory.getCode()))
147             {// 修改了code
148                 Map paramMap = new HashMap();
149                 paramMap.put("code", category.getCode());
150                 List<ActivitiCategory> temps =
151                     this.activitiCategoryDao.queryEntitysByMap(
152                         "from ActivitiCategory where id!="
153                         + tempCategory.getCategoryId()
154                         + " and code=:code and isDelete=0",
155                         paramMap);
156                 if (temps.size() > 0)
157                 {
158                     throw new ServiceException("系统已经存在修改的编码！");
159                 }
160                 this.createBpmFolder(category);
161             }
162             
163             tempCategory.setName(category.getName());
164             tempCategory.setParentId(category.getParentId());
165             tempCategory.setCode(category.getCode());
166             tempCategory.setSort(category.getSort());
167             tempCategory.setDescription(category.getDescription());
168             activitiCategoryDao.updateActivitiCategory(tempCategory);
169             
170         }
171         catch (ServiceException e)
172         {
173             throw new ServiceException(e.getMessageKey(),e);
174         }
175         catch (Exception e)
176         {
177             throw new ServiceException(e);
178         }
179     }

184     @Override
185     public ActivitiCategory getActivitiTopCategory(Integer categoryId)
186     {
187     	int pk_activiti_category_id = 0;
188     	int firstId = categoryId;
189     	do{
190     		String hql = "from ActivitiCategory a where a.isDelete = 0 and a.categoryId = "+firstId;
191     		List<ActivitiCategory> list = this.activitiCategoryDao.queryEntitys(hql);
192     		//如果list.size>0并且父亲ID = 0，说明已经是跟节点
193     		if(list.size()>0){
194     			if(list.get(0).getParentId()==0){
195         			pk_activiti_category_id = list.get(0).getCategoryId();
196         		}else{
197         			firstId = list.get(0).getParentId();
198         		}
199     		}else{
200     			break;
201     		} 
202     	}while(pk_activiti_category_id != 0);
203     	if(pk_activiti_category_id!=0){
204     		Map paramMap = new HashMap();
205             paramMap.put("isDelete", 0);
206             paramMap.put("categoryId", pk_activiti_category_id);
207             List<ActivitiCategory> categories =
208                 this.activitiCategoryDao.getActivitiCategory(paramMap);
209             if (categories.size() > 0)
210             {
211                 return categories.get(0);
212             }
213     	}
214 		return null;
215     }

217     @Override
218     public List<Map> getBPMType(Map map)
219         throws ServiceException
220     {
221         List<ActivitiCategory> datas = new ArrayList();
222         List<Map> treeNodes = new ArrayList();
223         Map tempMap = new HashMap();
224         try
225         {
226             String checkedFlag = (String)map.get("checkedFlag");
227             map.remove("checkedFlag");
228             map.put("isDelete", 0);
229             datas = this.activitiCategoryDao.getActivitiCategory(map);
230             for (ActivitiCategory c : datas)
231             {
232                 Map treeNode = new HashMap();
233                 treeNode.put("nodeId", c.getCategoryId());
234                 treeNode.put("parentId", c.getParentId());
235                 treeNode.put("text", c.getName());
236                 treeNode.put("type", c.getCode());
237                 treeNode.put("sort", c.getSort());
238                 treeNode.put("description", c.getDescription());
239                 tempMap.put("isDelete", 0);
240                 tempMap.put("parentId", c.getCategoryId());
241                 int count =
242                     this.activitiCategoryDao.
243                     queryTotalCount("from ActivitiCategory " +
244                     		" where isDelete=:isDelete " +
245                     		"  and parentId=:parentId",
246                         tempMap);
247                 if (count <= 0)
248                 {
249                     treeNode.put("leaf", true);
250                 }
251                 else
252                 {
253                     treeNode.put("leaf", false);
254                 }
255                 if ("true".equals(checkedFlag))
256                 {
257                     treeNode.put("checked", false);
258                 }
259                 
260                 treeNodes.add(treeNode);
261                 
262             }
263         }
264         catch (Exception e)
265         {
266             throw new ServiceException(e);
267         }
268         return treeNodes;
269     }

338     @Override
339     public Map createBpmFolder(ActivitiCategory category)
340     {
341         Map map = new HashMap();//相对路径，绝对路径
342         
343         List<ActivitiCategory> datas =
344             this.getLevelActivitiCategory(Integer.valueOf(category.getCategoryId()));
345        
346         StringBuffer path = new StringBuffer();
347         if (datas.size() > 0)
348         {
349             for (int i = datas.size() - 1; i >= 0; i--)
350             {
351                 path.append(datas.get(i).getCode()).append("/");
352             }
353             
354         }
355         path.append(category.getCode());
356         File pageTemp =
357             new File(ServletActionContext.getServletContext()
358                 .getRealPath("/")
359                 + SystemConstant.PAGE_FOLDER_ACTIVITI
360                 + File.separator + path.toString());
361         if (!pageTemp.exists() || !pageTemp.isDirectory())
362         {
363             pageTemp.mkdirs();
364         }
365         String floder =
366             ServletActionContext.getServletContext().getRealPath("/")
367                 + SystemConstant.UPLOAD_FOLDER_ACTIVITI + File.separator
368                 + path.toString();
369         File file = new File(floder);
370         if (!file.exists() || !file.isDirectory())
371         {
372             file.mkdirs();
373         }
374         map.put("relativeUrl", path.toString());
375         map.put("absoluteUrl", floder);
376         return map;
377     }
