145     /**
146      * 验证工作计划是否重复了
147      * 
148      * @Title validateWorkRulePlan
149      * @author dong.he 2014-8-22
150      * @param ruleId 规则id
151      * @param workTeamId 班组id
152      * @param roundId 班次id
153      * @param planDate 计划天数
154      * @Date 2014年9月2日
155      * @throws BusinessException
156      * @return WorkRulePlan 倒班计划对象
157      */
158     @Override
159     public WorkRulePlan validateWorkRulePlan(Integer ruleId,
160         Integer workTeamId, Integer roundId, String planDate)
161         throws BusinessException {
162         WorkRulePlan plan = null;
163         Map pram = new HashMap();
164         String hql =
165             " from WorkRulePlan wrp where 1=1 " + " and wrp.workTurnsRule.id="
166                 + ruleId + " and wrp.workRound.id=" + roundId
167                 + " and wrp.planDate=:planDate";
168         pram.put("planDate", DateUtil.stringToDate(planDate, "yyyy-dd-MM"));
169         List<WorkRulePlan> list =
170                 (List<WorkRulePlan>)baseDao.findPageByQuery(0,
171                 		Integer.MAX_VALUE,
172                     hql,
173                     pram);
174         
175         if (list != null && list.size() > 0) {
176             plan = list.get(0);
177         }
178         return plan;
179     }

220     /**
221      * 根据参数获得所有的已经生成的倒班计划
222      * 
223      * @Title getWorkRulePlanListByParams
224      * @author dong.he 2014-8-22
225      * @Date 2014年9月2日
226      * @param maps 倒班计划vo集合
227      * @throws BusinessException
228      * @return List<WorkRulePlanVo> 倒班计划vo集合
229      */
230     @Override
231     public ListVo<WorkRulePlanVo> getWorkRulePlanListByParams(
232         Map<String, String> paramsMap)
233         throws BusinessException {
234         ListVo<WorkRulePlanVo> pvo = new ListVo<WorkRulePlanVo>();
235         List<WorkRulePlanVo> vos = null;
236         WorkRulePlanVo vo = null;
237         Map pram = new HashMap();
238         String start = paramsMap.get("start");
239         String limit = paramsMap.get("limit");
240         String ruleId = paramsMap.get("ruleId");
241         String startDate = paramsMap.get("workRulePlanStartDate");
242         String endDate = paramsMap.get("workRulePlanEndDate");
243         
244         StringBuffer hql = new StringBuffer();
245         StringBuffer hqlCount = new StringBuffer();
246         hql.append(" from WorkRulePlan wrp where 1=1 ");
247         hqlCount.append(" select count(*) from WorkRulePlan wrp where 1=1 ");
248         if (StringUtils.isNotBlank(ruleId)) {
249             hql.append(" and wrp.workTurnsRule.id ="
250                 + NumberUtils.toInt(ruleId));
251             hqlCount.append(" and wrp.workTurnsRule.id ="
252                 + NumberUtils.toInt(ruleId));
253         }
254        
255         if (StringUtils.isNotBlank(startDate)) {
256             hql.append(" and wrp.planDate>=:startDate ");
257             hqlCount.append(" and wrp.planDate>=:startDate ");
258             pram.put("startDate", DateUtil.string2date(startDate));
259         }
260         if (StringUtils.isNotBlank(endDate)) {
261             hql.append(" and wrp.planDate<:endDate ");
262             hqlCount.append(" and wrp.planDate<:endDate ");
263             Date d = new Date(DateUtil.string2date(endDate).getTime()+24*60*60*1000);
264         	pram.put("endTime", d);
265         }
266         hql.append(" order by wrp.planDate ");
267         
268         // 获得班次的数量，便于取得数据的行数;
269         int roundCount =
270             baseDao.getTotalCount("select count(*) from WorkRound ",
271                 new HashMap<String, Object>());
272         int startCount = NumberUtils.toInt(start) * roundCount;
273         int limitCount = NumberUtils.toInt(limit) * roundCount;
274         List<WorkRulePlan> plans =
275             (List<WorkRulePlan>)baseDao.queryEntitysByPage(startCount,
276                 limitCount,
277                 hql.toString(),
278                 pram);
279         int count =
280             baseDao.getTotalCount(hqlCount.toString(),
281                 pram);
282         
283         if (plans != null && plans.size() > 0) {
284             vos = new ArrayList<WorkRulePlanVo>();
285             for (WorkRulePlan p : plans) {
286                 vo = new WorkRulePlanVo();
287                 vo.setCycleDays(p.getWorkTurnsRule().getCycleDays());
288                 vo.setId(p.getId());
289                 vo.setPlanDate(DateUtil.dateToString(p.getPlanDate(),
290                     "yyyy-MM-dd"));
291                 vo.setRoundId(p.getWorkRound().getId());
292                 vo.setRoundName(p.getWorkRound().getRoundName());
293                 vo.setRuleId(p.getWorkTurnsRule().getId());
294                 vo.setRuleName(p.getWorkTurnsRule().getName());
295                 vo.setWorkTeamId(p.getWorkTeam().getId());
296                 vo.setWorkTeamName(p.getWorkTeam().getTeamName());
297                 vo.setOrgId(p.getWorkTeam().getOrg().getOrgId());
298                 vo.setOrgName(p.getWorkTeam().getOrg().getOrgName());
299                 vos.add(vo);
300             }
301         }
302         pvo.setList(vos);
303         pvo.setTotalSize(count/roundCount);
304         return pvo;
305     }

309      * 删除指定时间段内的倒班计划
310      * 
311      * @Title deleteTurnsPlanByDate
312      * @author dong.he
313      * @date 2014年10月8日
314      * @param startDate 开始时间
315      * @param endDate 结束时间
316      * @throws BusinessException
317      */
318     @Override
319     public void deleteTurnsPlanByDate(String ruleId, String startDate, String endDate)
320         throws BusinessException {
321     	
322     	Map<String,Object> pram = new HashMap<String,Object>();
323         if (StringUtils.isNotBlank(startDate) && StringUtils.isNotBlank(endDate)
324             && StringUtils.isNotBlank(ruleId)) {
325             // 新建删除倒班计划的HQL语句
326             StringBuffer hql = new StringBuffer(" delete from WorkRulePlan wrp where ");
327             hql.append(" wrp.workTurnsRule.id = ");
328             hql.append(ruleId);
329             hql.append(" and wrp.planDate >= :startDate ");
330             hql.append(" and wrp.planDate <= :endDate ");
331             pram.put("startDate", DateUtil.string2date(startDate));
332             pram.put("endDate", DateUtil.string2date(endDate));
333             baseDao.executeHql(hql.toString(),pram);
334         }
